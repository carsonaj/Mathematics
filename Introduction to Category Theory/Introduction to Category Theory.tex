	\begin{defn} \ld{12001}
		Let $\MC_0$, $\MC_1$ be classes and $\dom, \cod : \MC_1 \rightarrow \MC_0$. Define $\MC_1^{(2)}\subset \MC_1 \times \MC_1$ by $\MC_1^{(2)} \defeq \{(g,f) \in \MC_1 \times \MC_1 : \dom(g) = \cod(f) \}$. Let $\circ: \MC_1^{(2)} \rightarrow \MC_1$. Then $(\MC_0, \MC_1, \dom, \cod, \circ)$ is said to be a \textbf{category} if 
		\begin{enumerate}
			\item \tbf{(Composition):} \\ 
			for each $(g,f) \in \MC_1^{(2)}$, $\dom(g \circ f) = \dom(f)$ and $\cod(g \circ f) = \cod(g)$
			\item \tbf{(Associativity):} \\ 
			for each $f,g,h \in \MC_1$, if $(g,f), (h,g) \in \MC_1^{(2)}$, then 
			$$(h \circ g) \circ f = h \circ (g \circ f)$$  
			\item \tbf{(Identity):} \\ 
			for each $X \in \MC_0$, there exists $e \in \MC_1$ such that $\dom(e) = X$, $\cod(e) = X$ and for each $f, g \in \MC_1$, if $\dom(f) = X$ and $\cod(g) = X$, then 
			$$f \circ e = f \text{ and } e \circ g = g$$ 
		\end{enumerate}
	\end{defn}
	
	\begin{defn}
		Let $\MC$ be a category. Write $\MC = (\MC_0, \MC_1, \dom, \cod, \circ)$.
		\begin{itemize}
			\item We define the
			\begin{itemize}
				\item \textbf{objects of $\MC$}, denoted $\Obj(\MC)$, by $\Obj(\MC) \defeq \MC_0$
				\item \textbf{morphisms of $\MC$}, denoted $\Hom$, by $\Hom_{\MC} \defeq \MC_1$ 
				\item \textbf{domain function of $\MC$}, denoted $\dom_{\MC}$, by $\dom_{\MC} \defeq \dom$
				\item \textbf{codomain function of $\MC$}, denoted $\cod_{\MC}$, by $\cod_{\MC} \defeq \cod$
				\item \textbf{composition operator of $\MC$}, denoted $\circ_{\MC}$, by $\circ_{\MC} \defeq \circ$
			\end{itemize}
			\item For $X, Y \in \Obj(\MC)$, we define the \textbf{Morphisms of $\MC$ from $X$ to $Y$}, denoted $\Hom_{\MC}(X, Y)$, by $\Hom_{\MC}(X, Y) \defeq \{f \in \Hom(\MC): \dom_{\MC}(f) = X \text{ and } \cod_{\MC}(f) = Y\}$.
		\end{itemize}
	\end{defn}

	\begin{note}
		When the context is clear, we write $\dom$, $\cod$ and $g \circ f$ in place of $\dom_{\MC}$, $\cod_{\MC}$ and $g \circ_{\MC} f$ respectively.
	\end{note}

	\begin{note} 
		We typically define a category $\MC$ by specifying 
		\begin{itemize}
			\item $\Obj(\MC)$
			\item for $X,Y \in \Obj(\MC)$, the class $\Hom_{\MC}(X, Y)$
			\item for $X,Y,Z \in \Obj(\MC)$, $f \in \Hom_{\MC}(X, Y)$ and $g \in \Hom_{\MC}(Y, Z)$, the composite morphism $g \circ f \in \Hom_{\MC}(X,Z)$.
		\end{itemize}
		and then show 
		\begin{itemize}
			\item well-definedness of composition
			\item associativity of composition 
			\item existence of identities 
		\end{itemize}
	\end{note}
	
	\begin{ex} \lex{15001} \textbf{Uniqueness of Identities:} \\
		Let $\MC$ be a category. Then for each $A \in \Obj(\MC)$, there exists a unique $e \in \Endo_{\MC}(A)$ such that for each $B \in \Obj(\MC)$, $f \in \Hom_{\MC}(A, B)$ and $g \in \Hom_{\MC}(B, A)$, $f \circ e = f$ and $e \circ g = g$.
	\end{ex}
	
	\begin{proof}
		Let $A \in \Obj(\MC)$. 
		\begin{itemize}
			\item \tbf{Existence:} \\
			Since $\MC$ is a category, by definition there exists $e \in \Hom_{\MC}(A, A)$ such that for each $B \in \Obj(\MC)$, $f \in \Hom_{\MC}(A, B)$ and $g \in \Hom_{\MC}(B, A)$, $f \circ e = f$ and $e \circ g = g$. 
			\item \tbf{Uniqueness: } \\
			Let $e_1, e_2 \in \Hom_{\MC}(A, A)$. Suppose that for each $B \in \Obj(\MC)$, $f \in \Hom_{\MC}(A, B)$ and $g \in \Hom_{\MC}(B, A)$, $f \circ e_1 = f$, $f \circ e_2 = f$, $e_1 \circ g = g$ and $e_2 \circ g = g$. Then 
			\begin{align*}
				e_1
				& = e_2 \circ e_1 \\
				& = e_2.
			\end{align*}
		\end{itemize}
	\end{proof}
	
	\begin{defn}
		Let $\MC$ be a category and $A \in \Obj(\MC)$. We define the \tbf{identity morphism on $A$}, denoted $\id_A^{\MC} \in \Hom_{\MC}(A, A)$, to be the unique $e \in \Hom_{\MC}(A, A)$ such that for each $B \in \Obj(\MC)$, $f \in \Hom_{\MC}(A, B)$ and $g \in \Hom_{\MC}(B, A)$, $f \circ e = f$ and $e \circ g = g$.
	\end{defn}
	
	\begin{note}
		When the context is clear, we write $\id_A$ in place of $\id_A^{\MC}$.
	\end{note}
	
	\begin{defn} \ld{12002}
		Let $\MC$ be a category. Then $\MC$ is said to be 
		\begin{itemize}
			\item \textbf{small} if $\Obj(\MC)$ and $\Hom_{\MC}$ are sets
			\item \textbf{locally small} if for each $A,B \in \Obj(\MC)$, $\Hom_{\MC}(A,B)$ is a set 
		\end{itemize}
	\end{defn}
	
	\begin{ex} \lex{12003}
		Let $\MC$ be a category. If $\MC$ is small, then $\MC$ is a set. 
	\end{ex}
	
	\begin{proof}
		Suppose that $\MC$ is small. Then $\Obj(\MC)$ and $\Hom_{\MC}$ are sets. Then $\MP(\Obj(\MC))$, $\MP(\Hom_{\MC})$ and $\Obj(\MC)^{\Hom_{\MC}}$ are sets. Hence $\Obj(\MC) \times \Hom_{\MC} \times \Obj(\MC)^{\Hom_{\MC}} \times \Obj(\MC)^{\Hom_{\MC}}$ is a set. By definition, $\MC = (\Obj(\MC), \Hom_{\MC}, \dom, \cod) \in \Obj(\MC) \times \Hom_{\MC} \times \Obj(\MC)^{\Hom_{\MC}} \times \Obj(\MC)^{\Hom_{\MC}}$. By definition, $\MC$ is a set. 
	\end{proof}
	
	\begin{ex} \lex{12004}
		There exists a class $A$ such that $\MC \in A$ iff $\MC$ is a small category. 
	\end{ex}
	
	\begin{proof}
		\rex{12003} implies that for each category $\MC$, $\MC$ is small implies that $\MC$ is a set. Define $\phi$ by $$\phi(\MC): \MC \text{ is a small category} $$ 
		Then \rax{11002} implies that there exists a class $A$ such that $\MC \in A$ iff $\MC$ is a small category.  
	\end{proof}

	\subsection{Opposite Categories}

	\begin{defn}  \ld{12004}
		Let $\MC$ be a category, we define the \text{dual of $\MC$} or the \textbf{opposite of $\MC$}, denoted $\op{\MC}$, by 
		\begin{itemize}
			\item $\Obj(\op{\MC}) \defeq \Obj(\MC)$
			\item for $X,Y \in \Obj(\op{\MC})$, $\Hom_{\op{\MC}}(X,Y) \defeq \Hom_{\MC}(Y,X)$
			\item for $X, Y, Z \in \Obj(\op{\MC})$ and $f \in \Hom_{\op{\MC}}(X, Y), g \in \Hom_{\op{\MC}}(Y,Z)$, $g \circ_{\op{\MC}} f \defeq f \circ_{\MC} g$
		\end{itemize}
	\end{defn}

	\begin{ex}  \lex{12005}
		Let $\MC$ be a category. Then $\op{\MC}$ is a category.
	\end{ex}
	
	\begin{proof}\
		\begin{itemize}
			\item for $W,X,Y,Z \in \Obj(\MC)$, $f \in \Hom_{\op{\MC}}(W, X)$ and $g \in \Hom_{\op{\MC}}(X, Y)$ and $h \in \Hom_{\op{\MC}}(Y, Z)$. Then 
			\begin{align*}
				(h \circ_{\op{\MC}} g) \circ_{\op{\MC}} f 
				&= f \circ_{\MC}( h \circ_{\op{\MC}} g) \\
				&= f \circ_{\MC} (g \circ_{\MC} h) \\
				&= (f \circ_{\MC} g) \circ_{\MC} h \\
				&= h \circ_{\op{\MC}} (f \circ_{\MC} g) \\
				&= h \circ_{\op{\MC}} (g \circ_{\op{\MC}} f)
			\end{align*}
			So composition is associative.
			\item Let $X \in \Obj(\MC)$ and $f,g \in \Hom_{\op{\MC}}$. Suppose that $\dom(f) = X$ and $\cod(g) = X$ 
			Then 
			\begin{align*}
				f  \circ_{\op{\MC}} \id_X
				&= \id_X  \circ_{\MC} f \\
				&= f
			\end{align*}
			and 
			\begin{align*}
				\id_X  \circ_{\op{\MC}} g 
				&= g \circ_{\MC} \id_X \\
				&= g
			\end{align*}
		So $(\id_X)_{\op{\MC}} =  (\id_X)_{\MC}$.
		\end{itemize}
	\end{proof}

	\begin{defn} \ld{12006}
		\tcr{rework make $\Obj(\MC/X) \defeq \{(A, f): A \in \Obj(\MC) \text{ and } f \in \Hom_{\MC}(A, X)\}$}
		Let $\MC$ be a category and $X \in \Obj(\MC)$. We define the \textbf{slice category of $\MC$ over $X$}, denoted $\MC / X$, by
		\begin{itemize}
			\item $\Obj(\MC / X) \defeq \{(A, f): A \in \Obj(\MC) \text{ and } f \in \Hom_{\MC}(A, X)\}$
			\item for $(A, f),( B, g) \in \Obj(\MC / X)$, 
			$$\Hom_{\MC / X}((A,f), (B,g)) = \{\al \in \Hom_{\MC}(A, B) : g \circ_{\MC} \al = f \}$$
			i.e. for $f \in \Hom_{\MC}(A, X)$ and $g \in \Hom_{\MC}(B, X)$, $\al \in \Hom_{\MC / X}((A, f), (B, g))$ iff the following diagram commutes: 
			\[ 
			\begin{tikzcd}
				A \arrow[rr, "\al"] \arrow[dr, "f"'] 	
				& & B  \arrow[dl, "g"] \\
				& X 
			\end{tikzcd}
			\]
			\item for $(A, f), (B, g), (C, h) \in \Obj(\MC / X)$, $\al \in \Hom_{\MC / X}((A, f), (B,g))$ and $\be \in \Hom_{\MC / X}((B,g), (C,h))$, 
			$$\be \circ_{\MC / X} \al \defeq \be \circ_{\MC} \al$$
		\end{itemize}
	\end{defn}
	
	\begin{ex}  \lex{12007}
		Let $\MC$ be a category and $X \in \Obj(\MC)$. Then 
		\begin{enumerate}
			\item $\MC / X$ is a category,
			\item for each $(A,f) \in \Obj(\MC/X)$, $\id_{(A,f)}^{\MC/X} = \id_A^{\MC}$.
		\end{enumerate}
	\end{ex}

	\begin{proof}\
		\begin{enumerate}
			\item \begin{itemize}
				\item \tbf{Well-Definedness of Composition:} \\
				$(A, f), (B, g), (C, h) \in \Obj(\MC / X)$, $\al \in \Hom_{\MC / X}((A, f), (B,g))$ and $\be \in \Hom_{\MC / X}((B,g), (C,h))$. Then $g \circ_{\MC} \al  = f$ and $h \circ_{\MC} \be = g$, i.e. the following diagrams commute:
				\[ \begin{tikzcd}
					A \arrow[rr, "\al"] \arrow[dr, "f"'] 	
					&& B  \arrow[dl, "g"]  
					&& B \arrow[rr, "\be"] \arrow[dr, "g"'] 	
					&& C  \arrow[dl, "h"]\\
					& X 
					&& &&X
				\end{tikzcd}
				\]
				Therefore, we have that 
				\begin{align*}
					h \circ_{\MC} (\be \circ_{\MC} \al) 
					& = (h \circ_{\MC} \be) \circ_{\MC} \al \\
					& = g \circ_{\MC} \al \\
					& = f
				\end{align*}
				i.e. the following diagram commutes:
				\[ 
				\begin{tikzcd}
					A \arrow[rr, "\be \circ_{\MC} \al"] \arrow[dr, "f"'] 	
					& & C  \arrow[dl, "h"] \\
					& X 
				\end{tikzcd}
				\]
				which implies that 
				\begin{align*}
					\be \circ_{\MC / X} \al 
					& = \be \circ_{\MC} \al \\
					& \in \Hom_{\MC / X}((A,f), (C,h))
				\end{align*}
				and composition is well defined. 
				\item \tbf{Associativity:} \\
				Associativity of $\circ_{\MC / X}$ follows from associativity of $\circ_\MC$.
				\item \tbf{Existence of Identities:} \\
				Let $(A,f) \in \Obj(\MC / X)$ and $\al, \be \in \Hom_{\MC / X}$. Suppose that $\dom_{\MC / X}(\al) = (A,f)$ and $\cod_{\MC /X}(\be) = (A,f)$.
				\begin{itemize}
					\item Since $f \circ_{\MC} \id_A^{\MC} = f$, i.e. the following diagram commutes: 
					\[ \begin{tikzcd}
						A \arrow[rr, "\id_A^{\MC}"] \arrow[dr, "f"'] 	
						& & A  \arrow[dl, "f"] \\
						& X 
					\end{tikzcd}
					\]
					we have that $\id_A^{\MC} \in \Hom_{\MC / X}((A,f), (A,f))$. 
					\item Then 
					\begin{align*}
						\al \circ_{\MC /X} \id_A^{\MC}
						&= \al \circ_{C} \id_A^{\MC} \\
						&= \al
					\end{align*}
					and 
					\begin{align*}
						\id_A^{\MC} \circ_{\MC /X} \be
						&= \id_A^{\MC} \circ_{\MC} \be \\
						&= \be.
					\end{align*}
				\end{itemize}
				Since $(A,f) \in \Obj(\MC / X)$ and $\al, \be \in \Hom_{\MC / X}$ with $\dom_{\MC / X}(\al) = (A,f)$ and $\cod_{\MC /X}(\be) = (A,f)$ are arbitrary, we have that for each $(A,f) \in \Obj(\MC / X)$, there exists $e \in \Hom_{\MC / X}((A,f), (A,f))$ such that $\dom_{\MC/X}(e) = (A,f)$, $\cod_{\MC/X}(e) = (A,f)$ and for each $\al, \be \in \Hom_{\MC}$, if $\dom_{\MC/X}(\al) = (A,f)$ and $\cod_{\MC/X}(\be) = (A,f)$, then $\al \circ_{\MC/X} e = \al$ and $e \circ_{\MC/X} \be = \be$. 
			\end{itemize}
			Therefore $\MC/X$ is a category.
			\item \tcr{From above}, we have that for each $(A,f) \in \Obj(\MC/X)$, $\id_{(A,f)}^{\MC/X} = \id_A^{\MC}$.
		\end{enumerate}
	\end{proof}

		Let $\MC$ and $\MD$ be categories. Suppose that $\MC$ is small. Define $\phi$ by 
		$$\phi(F) : F : \MC \rightarrow \MD$$ 
		Then there exists a class $A$ such that for each set $F$, $F \in A$ iff $\phi(F)$. Let $F$ be a class. Suppose that $F \in A$. By \rd{11001}, $F$ is a set. Since $F$ is a set and $F \in A$, we have that $\phi(F)$. Hence $F: \MC \rightarrow \MD$. \\
		Conversely, suppose that $F: \MC \rightarrow \MD$. \rex{13008.1} implies that $F$ is a set. Since $F$ is a set and $\phi(F)$ is true, we have that $F \in A$. 
	\end{proof}
	
	\begin{defn} \ld{13009}
		We define $\tbf{Cat}$ by 
		\begin{itemize}
			\item $\Obj(\tbf{Cat}) = \{\MC: \MC \text{ is a small category}\}$.
			\item for $\MC,\MD \in \Obj(\tbf{Cat})$, 
			$$\Hom_{\tbf{Cat}}(\MC,\MD) = \{F : F: \MC \rightarrow \MD \}$$
			\item for $\MC,\MD, \ME \in \Obj(\tbf{Cat})$, $F \in \Hom_{\tbf{Cat}}(\MC,\MD)$ and $G \in \Hom_{\tbf{Cat}}(\MD, \ME)$, $$G \circ_{\tbf{Cat}} F = G \circ F$$
		\end{itemize}
	\end{defn}


	\begin{ex}  \lex{13010}
		We have that \tbf{Cat} is 
		\begin{enumerate}
			\item a category
			\item locally small
		\end{enumerate} 
	\end{ex}

	\begin{proof}\
		\begin{enumerate}
			\item \rex{13007} implies that composition is well defined. \rex{13008} implies that composition is associative. \rex{13005} and \rex{13006} imply the existence of identities.
			\item Let $\MC, \MD \in \Obj(\tbf{Cat})$ and $F \in \Hom_{\tbf{Cat}}(\MC, \MD)$. \rd{12002} implies that $\Obj(\MC)$, $\Obj(\MD)$, $\Hom_{\MC}$ and $\Hom_{\MD}$ are sets. Then $\Obj(\MD)^{\Obj(\MC)}$ and $\Hom_{\MD}^{ \Hom_{\MC}}$ are sets. Hence $ \Obj(\MD)^{\Obj(\MC)} \times  \Hom_{\MD}^{ \Hom_{\MC}}$ is a set. Let $F \in \Hom_{\tbf{Cat}}(\MC, \MD)$. Then there exist $F_0 \in \Obj(\MD)^{\Obj(\MC)} $ and $F_1 \in \Hom_{\MD}^{ \Hom_{\MC}}$ such that $F = (F_0, F_1)$. Therefore $F \in  \Obj(\MD)^{\Obj(\MC)} \times  \Hom_{\MD}^{ \Hom_{\MC}}$. Since $F \in \Hom_{\tbf{Cat}}(\MC, \MD)$ is arbitrary, 
			\begin{align*}
				\Hom_{\tbf{Cat}}(\MC, \MD) 
				& \subset \Obj(\MD)^{\Obj(\MC)} \times  \Hom_{\MD}^{ \Hom_{\MC}}
			\end{align*}
			which implies that $\Hom_{\tbf{Cat}}(\MC, \MD)$ is a set. Therefore, $\tbf{Cat}$ is locally small.
		\end{enumerate} 
	\end{proof}
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	















	\subsection{Comma Categories}

	
	\begin{defn} \ld{13011}
		Let $\MA$, $\MB$, $\MC$ be a categories and $S: \MA \rightarrow \MC$, $T: \MB \rightarrow \MC$ functors. We define the \textbf{comma category of $S$ to $T$}, denoted $(S \doar T)$, by 
		\begin{itemize}
			\item $\Obj(S \doar T) = \{(A, B, h): A \in \Obj(\MA), B \in \Obj(\MB) \text{, and } h \in \Hom_{\MC}(S(A), T(B))\}$
			\item For $(A_1, B_1, h_1), (A_2, B_2, h_2) \in \Obj(S \doar T)$, 
			\begin{align*}
				& \Hom_{(S \doar T)}((A_1, B_1, h_1), (A_2, B_2, h_2)) = \\
				& \quad \quad \{(\al, \be): \al \in \Hom_{\MA}(A_1, A_2) \text{, } \be \in \Hom_{\MB}(B_1, B_2) \text{ and } T(\be) \circ_{\MC} h_1 = h_2 \circ_{\MC} S(\al)\}
			\end{align*}
			i.e. for $(A_1, B_1, h_1), (A_2, B_2, h_2) \in \Obj(S \doar T)$, $\al \in \Hom_{\MA}(A_1, A_2)$ and $\be \in \Hom_{\MB}(B_1, B_2)$, $(\al, \be) \in \Hom_{(S \doar T)}((A_1, B_1, h_1), (A_2, B_2, h_2))$ iff the following diagram commutes:
			\[ 
			\begin{tikzcd}
				S(A_1) \arrow[r, "S(\al)"] \arrow[d, "h_1"'] & S(A_2)  \arrow[d, "h_2"] \\
				T(B_1) \arrow[r, "T(\be)"']                 & T(B_2)
			\end{tikzcd}
			\]
			\item For 
			\begin{itemize}
				\item $(A_1, B_1, h_1), (A_2, B_2, h_2), (A_3, B_3, h_3) \in \Obj(S \doar T)$
				\item $(\al_{12}, \be_{12}) \in \Hom_{(S \doar T)}((A_1, B_1, h_1), (A_2, B_2, h_2))$
				\item $(\al_{23}, \be_{23}) \in \Hom_{(S \doar T)}((A_2, B_2, h_2), (A_3, B_3, h_3))$
			\end{itemize}
			we define 
			$$(\al_{23}, \be_{23}) \circ_{(S \doar T)} (\al_{12}, \be_{12}) = (\al_{23} \circ_{\MA} \al_{12}, \be_{23} \circ_{\MB} \be_{12})$$
		\end{itemize}
	\end{defn}
	
	\begin{ex} \lex{13013}
		Let $\MA$, $\MB$, $\MC$ be a categories and $S: \MA \rightarrow \MC$, $T: \MB \rightarrow \MC$ functors. Then $(S \doar T)$ is a category.
	\end{ex}
	
	\begin{proof}\
		\begin{itemize}
			\item \textbf{well-definedness of composition:} \\
			Let 
			\begin{itemize}
				\item $(A_1, B_1, h_1), (A_2, B_2, h_2), (A_3, B_3, h_3) \in \Obj(S \doar T)$
				\item $(\al_{12}, \be_{12}) \in \Hom_{(S \doar T)}((A_1, B_1, h_1), (A_2, B_2, h_2))$ 
				\item $(\al_{23}, \be_{23}) \in \Hom_{(S \doar T)}((A_2, B_2, h_2), (A_3, B_3, h_3))$
			\end{itemize}
			By definition, $\al_{12} \in \Hom_{\MA}(A_1, A_2)$, $\al_{23} \in \Hom_{\MA}(A_2, A_3)$, $\be_{12} \in \Hom_{\MB}(B_1, B_2)$, $\be_{23} \in \Hom_{\MB}(B_2, B_3)$, $T(\be_{12}) \circ_{\MC} h_1 = h_2 \circ S(\al_{12})$ and $T(\be_{23}) \circ_{\MC} h_2 = h_3 \circ_{\MC} S(\al_{23})$, \\
			i.e. the following diagram commutes:
			\[ 
			\begin{tikzcd}
				S(A_1) \arrow[r, "S(\al_{12})"] \arrow[d, "h_1"'] & S(A_2)  \arrow[d, "h_2"] \arrow[r, "S(\al_{23})"] 
				& S(A_3) \arrow[d, "h_3"]\\
				T(B_1) \arrow[r, "T(\be_{12})"']                  & T(B_2)                   \arrow[r, "T(\be_{23})"'] 
				& T(B_3) 
			\end{tikzcd}
			\]
			Then $\al_{23} \circ_{\MA} \al_{12} \in \Hom_{\MA}(A_1, A_3)$, $\be_{23} \circ_{\MB} \be_{12} \in \Hom_{\MB}(B_1, B_3)$ and 
			\begin{align*}
				T(\be_{23} \circ_{\MB} \be_{12}) \circ_{\MC} h_1
				& = (T(\be_{23}) \circ_{\MC} T(\be_{12})) \circ_{\MC} h_1 \\
				& = T(\be_{23}) \circ_{\MC} (T(\be_{12}) \circ_{\MC} h_1) \\
				& = T(\be_{23}) \circ_{\MC} (h_2 \circ_{\MC} S(\al_{12})) \\
				& = (T(\be_{23}) \circ_{\MC} h_2) \circ_{\MC} S(\al_{12}) \\
				& = (h_3 \circ_{\MC} S(\al_{23})) \circ_{\MC} S(\al_{12}) \\
				& = h_3 \circ_{\MC} (S(\al_{23}) \circ_{\MC} S(\al_{12})) \\
				& = h_3 \circ_{\MC} S(\al_{23} \circ_{\MA} \al_{12})
			\end{align*}
			i.e. the following diagram commutes:
			\[ 
			\begin{tikzcd}
				S(A_1) \arrow[rr, "S(\al_{23} \circ_{\MA} \al_{12})"] \arrow[d, "h_1"']&  & S(A_3)  \arrow[d, "h_3"] \\
				T(B_1) \arrow[rr, "T(\be_{23} \circ_{\MB} \be_{12})"']                       &  & T(B_3)
			\end{tikzcd}
			\]
			Hence $(\al_{23} \circ_{\MA} \al_{12}, \be_{23} \circ_{\MB} \be_{12}) \in \Hom_{(S \doar T)}((A_1, B_1, h_1), (A_3, B_3, h_3))$ and composition is well defined. \\
			\item \textbf{associativity of composition:} \\
			Let 
			\begin{itemize}
				\item $(A_1, B_1, h_1), (A_2, B_2, h_2), (A_3, B_3, h_3), (A_4, B_4, h_4) \in \Obj(S \doar T)$
				\item $(\al_{12}, \be_{12}) \in \Hom_{(S \doar T)}((A_1, B_1, h_1), (A_2, B_2, h_2))$ 
				\item $(\al_{23}, \be_{23}) \in \Hom_{(S \doar T)}((A_2, B_2, h_2), (A_3, B_3, h_3))$
				\item $(\al_{34}, \be_{34}) \in \Hom_{(S \doar T)}((A_3, B_3, h_3), (A_4, B_4, h_4))$
			\end{itemize}
			Then 
			\begin{align*}
				[ (\al_{34}, \be_{34}) \circ_{(S \doar T)} (\al_{23}, \be_{23}) ] \circ_{(S \doar T)} (\al_{12}, \be_{12})
				& = (\al_{34} \circ_{\MA} \al_{23}, \be_{34} \circ_{\MB} \be_{23}) \circ_{(S \doar T)} (\al_{12}, \be_{12}) \\
				& = ([\al_{34} \circ_{\MA} \al_{23}] \circ_{\MA} \al_{12}, [\be_{34} \circ_{\MB} \be_{23}] \circ_{\MB} \be_{12}) \\
				& =  (\al_{34} \circ_{\MA} [\al_{23} \circ_{\MA} \al_{12}], \be_{34} \circ_{\MB} [\be_{23} \circ_{\MB} \be_{12}]) \\
				& = (\al_{34}, \be_{34}) \circ_{(S \doar T)} (\al_{23} \circ_{\MA} \al_{12}, \be_{23} \circ_{\MB} \be_{12}) \\
				& = (\al_{34}, \be_{34}) \circ_{(S \doar T)} [(\al_{23}, \be_{23}) \circ_{(S \doar T)} (\al_{12}, \be_{12})]
			\end{align*}
			So composition is associative. \\
			\item \textbf{existence of identities:} \\
			Let
			\begin{itemize}
				\item $(A_1, B_1, h_1), (A_2, B_2, h_2), \in \Obj(S \doar T)$
				\item $(\al, \be) \in \Hom_{(S \doar T)}((A_1, B_1, h_1), (A_2, B_2, h_2))$ 
			\end{itemize}
			By definition, 
			\begin{itemize}
				\item $\al \in \Hom_{\MA}(A_1, A_2)$, $\be \in \Hom_{\MB}(B_1, B_2)$
				\item $h_1 \in \Hom_{\MC}(S(A_1), T(B_1))$, $h_2 \in \Hom_{\MC}(S(A_2), T(B_2))$ 
				\item $T(\be) \circ h_1 = h_2 \circ S(\al)$
			\end{itemize}
			Since $\id_{A_1} \in \Hom_{\MA}(A_1, A_1)$, $\id_{B_1} \in \Hom_{\MB}(B_1, B_1)$, and
			\begin{align*}
				T(\id_{B_1}) \circ_{\MC}  h_1 
				& = \id_{T(B_1)} \circ_{\MC} h_1 \\
				& = h_1 \\
				& = h_1 \circ_{\MC} \id_{S(A_1)} \\
				& = h_1 \circ_{\MC} S(\id_{A_1}) 
			\end{align*}
			i.e. the following diagram commutes:
			\[
			\begin{tikzcd}
				S(A_1) \arrow[r, "S(\id_{A_1})"] \arrow[d, "h_1"']  & S(A_1)  \arrow[d, "h_1"] \\
				T(B_1) \arrow[r, "T(\id_{B_1})"']                   & T(B_1)
			\end{tikzcd}
			\]
			we have that $(\id_{A_1}, \id_{B_1}) \in \Hom_{(S \doar T)}((A_1, B_1, h_1), (A_1, B_1, h_1))$. Similarly $(\id_{A_2}, \id_{B_2}) \in \Hom_{(S \doar T)}((A_2, B_2, h_2), (A_2, B_2, h_2))$. Therefore
			\begin{align*}
				(\al, \be) \circ_{(S \doar T)} (\id_{A_1}, \id_{B_1}) 
				& = (\al \circ_{\MA} \id_{A_1}, \be \circ_{\MB} \id_{B_1}) \\
				& = (\al, \be)
			\end{align*}
			and 
			\begin{align*}
				(\id_{A_2}, \id_{B_2}) \circ_{(S\doar T)} (\al, \be) 
				& = (\id_{A_2} \circ_{\MA} \al ,  \id_{B_2} \circ_{\MB} \be ) \\
				& = (\al, \be)
			\end{align*}
			Since $(A_1, B_1, h_1), (A_2, B_2, h_2), \in \Obj(S \doar T)$ and\\ $(\al, \be) \in \Hom_{(S \doar T)}((A_1, B_1, h_1), (A_2, B_2, h_2))$ are arbitrary, we have that for each $(A, B, h) \in \Obj(S \doar T)$, $\id_{(A, B, h)} = (\id_A, \id_B)$.
		\end{itemize}
	\end{proof}

	\section{Natural Transformations}
	
	\subsection{Introduction}
	
	\begin{defn} \ld{14001}
		Let $\MC$ and $\MD$ be categories, $F, G: \MC \rightarrow \MD$ and $ \al : \Obj(\MC) \rightarrow  \Hom_{\MD}$. Then $\al$ is said to be a \textbf{natural transformation from $F$ to $G$}, denoted $\al: F \Rightarrow G$, if
		\begin{enumerate}
			\item for each $A \in \Obj(\MC)$, $\al_A \in \Hom_{\MD}(F(A), G(A))$
			\item for each $A, B \in \Obj(\MC)$ and $f \in \Hom_{\MC}(A,B)$, $G(f) \circ \al_A = \al_B \circ F(f)$, i.e. the following diagram commutes: 
			\[ 
			\begin{tikzcd}
				F(A)  \arrow[r, "\al_A"]  \arrow[d, "F(f)"']  & G(A)   \arrow[d, "G(f)"]\\
				F(B) \arrow[r, "\al_B"] &  G(B) \\
			\end{tikzcd}
			\]
		\end{enumerate}
	\end{defn}






	
	
	
	
	
	
	
	
	
	
	\subsection{Category of Functors}
	
	\begin{defn} \ld{14002}
		Let $\MC$, $\MD$ be categories, $F, G, H:\MC \rightarrow \MD$ functors and $\al: F \Rightarrow G$, $\be : G \Rightarrow H$ natural transformations. We define the \textbf{composition of $\be$ with $\al$}, denoted $\be \circ \al: F \Rightarrow H$, by 
		$$(\be \circ \al)_{A} = \be_A \circ \al_A$$ 
	\end{defn}
	
	\begin{ex}  \lex{14003}
		Let $\MC$, $\MD$ be categories, $F, G, H:\MC \rightarrow \MD$ functors and $\al: F \Rightarrow G$, $\be : G \Rightarrow H$ natural transformations. Then $ \be \circ \al: F \Rightarrow H$ is a natural transformation.
	\end{ex}
	
	\begin{proof}\
		\begin{enumerate}
			\item Let $A \in \Obj(\MC)$. Since $\al_A \in \Hom_{\MD}(F(A), G(A))$ and $\be_A \in \Hom_{\MD}(G(A), H(A))$, we have that 
			\begin{align*}
				(\be \circ \al)_A 
				& = \be_A \circ \al_A \\
				& \in \Hom_{\MD}(F(A), H(A))
			\end{align*}
			\item Let $A,B \in \Obj(\MC)$ and $f \in \Hom_{\MC}(A,B)$. Since $\al:F \Rightarrow G$ and $\be: G \Rightarrow H$, $G(f) \circ \al_A = \al_B \circ F(f)$ and $H(f) \circ \be_A = \be_B \circ G(f)$. Therefore 
			\begin{align*}
				H(f) \circ (\be \circ \al)_A
				& = H(f) \circ (\be_A \circ \al_A) \\
				& = (H(f) \circ \be_A) \circ \al_A \\
				& = (\be_B \circ G(f)) \circ \al_A \\
				& = \be_B \circ (G(f)\circ \al_A) \\
				& = \be_B \circ (\al_B \circ F(f)) \\
				& = (\be_B \circ \al_B) \circ F(f) \\
				& = (\be \circ \al)_B \circ F(f)
			\end{align*}
		\end{enumerate}
		So $\be \circ \al: F \Rightarrow H$ is a natural transformation. 
	\end{proof}

	\begin{ex} \lex{14003.1}
		Let $\MC$, $\MD$ be categories, $F, G, H, K:\MC \rightarrow \MD$ functors and $\al: F \Rightarrow G$, $\be : G \Rightarrow H$ and $\gam: H \Rightarrow K$ natural transformations. Then $$(\gam \circ \be) \circ \al = \gam \circ (\be \circ \al)$$
	\end{ex}

	\begin{proof}
		Let $A \in \Obj(\MC)$. By definition,  
		\begin{align*}
			[(\gam \circ \be) \circ \al]_{A} 
			& = (\gam \circ \be)_A \circ \al_{A} \\
			& = (\gam_A \circ \be_A) \circ \al_A \\
			& = \gam_A \circ (\be_A \circ \al_A) \\
			& = \gam_A \circ (\be \circ \al)_A \\
			& = [\gam \circ (\be \circ \al)]_A \\
		\end{align*}
		Since $A \in \Obj(\MC)$ is arbitrary, 
		$$(\gam \circ \be) \circ \al = \gam \circ (\be \circ \al)$$
	\end{proof}

	\begin{defn} \ld{14003.2}
		Let $\MC$, $\MD$ be categories and $F: \MC \rightarrow \MD$. We define the \textbf{identity natural transformation from $F$ to $F$}, denoted $\id_{F}: F \Rightarrow F$, by 
		$$(\id_F)_A = \id_{F(A)}$$
	\end{defn}
	
	\begin{ex}  \lex{14003.3}
		Let $\MC$, $\MD$ be categories and $F: \MC \rightarrow \MD$. Then $\id_F: F \Rightarrow F$ is a natural transformation.
	\end{ex}
	
	\begin{proof}\
		\begin{enumerate}
			\item Let $A \in \Obj(\MC)$. Then 
			\begin{align*}
				(\id_F)_A 
				& = \id_{F(A)} \\
				& \in \Hom_{\MD}(F(A), F(A))
			\end{align*}
			\item Let $A,B \in \Obj(\MC)$, $f \in \Hom_{\MC}(A,B)$. Then 
			\begin{align*}
				F(f) \circ (\id_{F})_{A} 
				& = F(f) \circ \id_{F(A)} \\
				& = F(f) \\
				& = \id_{F(B)} \circ F(f) \\
				& = (\id_{F})_{B} \circ F(f)
			\end{align*}
		\end{enumerate}
	\end{proof}

	\begin{ex} \lex{14003.4}
		Let $\MC$, $\MD$ be categories, $F,G: \MC \rightarrow \MD$ and $\al : F \Rightarrow G$. Then 
		\begin{enumerate}
			\item $\id_{G} \circ \al = \al$

	\begin{defn} \ld{14008}
		Let $\MC$, $\MD$ be categories, $X, Y \in \Obj(\MD)$ and $f \in \Hom_{\MD}(X, Y)$. We define the \textbf{constant natural transformation on $\MC$ at $f$}, denoted $\del^{\MC}_f: \Del^{\MC}_X \Rightarrow \Del^{\MC}_Y$, by
		$$(\del^{\MC}_f)_A = f$$
	\end{defn}
	
	\begin{ex} \lex{14009}
		Let $\MC$, $\MD$ be categories, $X, Y \in \Obj(\MD)$ and $f \in \Hom_{\MD}(X, Y)$. Then $\del^{\MC}_f: \Del^{\MC}_X \Rightarrow \Del^{\MC}_Y $ is a natural transformation.
	\end{ex}
	
	\begin{proof} \
		\begin{enumerate}
			\item By definition, for each $A \in \Obj(\MC)$ $(\del^{\MC}_f)_A \in \Hom_{\MD}(\Del^{\MC}_X(A), \Del^{\MC}_Y(A))$.
			\item Let $A, B \in \Obj(\MC)$ and $g \in \Hom_{\MC}(A, B)$. Then 
			\begin{align*}
				\Del^{\MC}_Y(g) \circ (\del^{\MC}_f)_A 
				& = \id_Y \circ f \\
				& = f \\
				& = f \circ \id_X \\
				& = (\del^{\MC}_f)_B \circ \Del^{\MC}_X(g)  
			\end{align*}
			i.e. the following diagram commutes:
			\[ 
			\begin{tikzcd}[baseline= 7]
				\Del^{\MC}_X(A)  \arrow[r, "(\del^{\MC}_f)_A"]  \arrow[d, "\Del^{\MC}_X(g)"']  & \Del^{\MC}_Y(A)   \arrow[d, "\Del^{\MC}_Y(g)"]\\
				\Del^{\MC}_X(B) \arrow[r, "(\del^{\MC}_f)_B"] &  \Del^{\MC}_Y(B) \\
			\end{tikzcd}
			=
			\begin{tikzcd}[baseline= 7]
				X  \arrow[r, "f"]  \arrow[d, "\id_X"']  & Y   \arrow[d, "\id_Y"]\\
				X \arrow[r, "f"] &  Y \\
			\end{tikzcd}
			\]
		\end{enumerate}
		So $\del^{\MC}_f: \Del^{\MC}_X \Rightarrow \Del^{\MC}_Y$ is a natural transformation.
	\end{proof}
	 
	\begin{ex} \lex{14010}
		Let $\MC, \MD$ be categories, $X, Y, Z \in \Obj(\MD)$, $f \in \Hom_{\MD}(X,Y)$ and $g \in \Hom_{\MD}(Y,Z)$. Then $\del^{\MC}_{g \circ f} = \del^{\MC}_g \circ \del^{\MC}_f$.

		\end{align*}
		Since $A \in \Obj(\MC)$ is arbitrary, $\del^{\MC}_{g \circ f} = \del^{\MC}_g \circ \del^{\MC}_f$.
	\end{proof}

	\begin{defn} \ld{def:basics:contructs:subcats:0001}
		Let $\MC$ be a category. 
		\begin{itemize}
			\item Let $\MD$ be categories. Then $\MD$ is said to be a \tbf{subcategory of $\MC$}, denoted $\MD \subset \MC$, if 
			\begin{enumerate}
				\item $\Obj(\MD) \subset \Obj(\MC)$
				\item for each $A, B \in \Obj(\MD)$, $\Hom_{\MD}(A,B) \subset \Hom_{\MC}(A, B)$
				\item for each $A,B,C \in \Obj(\MD)$, $f \in \Hom_{\MD}(A, B)$ and $g \in \Hom_{\MD}(B, C)$, $g \circ_{\MD} f = g \circ_{\MC} f$
				\item for each $A \in \Obj(\MD)$, $\id_A^{\MD} = \id_A^{\MC}$
			\end{enumerate}

		\end{itemize}
	\end{defn}

		\begin{itemize}
			\item $\dom_{\MC}(\MD_1), \cod_{\MC}(\MD_1) \subset \MD_0$,
			\item $\circ_{\MC}(\MD_1^{(2)}) \subset \MD_1$,
			\item for each $X \in \MD_0$, $\id_X^{\MC} \in \MD_1$,
		\end{itemize}

			\item $(\MD_0, \MD_1, \dom_{\MC}|_{\MD_1}, \cod_{\MC}|_{\MD_1}, \circ_{\MC}|_{\MD_1^{(2)}})$ is a category,
			\item $(\MD_0, \MD_1, \dom_{\MC}|_{\MD_1}, \cod_{\MC}|_{\MD_1}, \circ_{\MC}|_{\MD_1^{(2)}})$ is a subcategory of $\MC$.
		\end{enumerate}
	\end{ex}
	
	\begin{proof}
		Set $\dom_{\MD} \defeq \dom_{\MC}|_{\MD_1}$, $\cod_{\MD} \defeq \cod_{\MC}|_{\MD_1}$, $\circ_{\MD} \defeq \circ_{\MC}|_{\MD_1^{(2)}}$ and $\MD \defeq (\MD_0, \MD_1, \dom_{\MC}|_{\MD_1}, \cod_{\MC}|_{\MD_1}, \circ_{\MC}|_{\MD_1^{(2)}})$.
		\begin{enumerate}
			\item 
			\begin{enumerate}
				\item \tbf{(Composition:)} \\
				For each $(g,f) \in \MD_1^{(2)}$, 
				\begin{align*}
					\dom_{\MD}(g \circ_{\MD} f)
					& = \dom_{\MC}(g \circ_{\MC}) \\
					& = \dom_{\MC}(f) \\
					& = \dom_{\MD}(f).
				\end{align*}
				and 
				\begin{align*}
					\cod_{\MD}(g \circ_{\MD} f)
					& = \cod_{\MC}(g \circ_{\MC}) \\
					& = \cod_{\MC}(g) \\
					& = \cod_{\MD}(g).
				\end{align*}
				\item \tbf{(Associativity:)} \\
				For each $f,g,h \in \MD_1$, if $(g,f), (h,g) \in \MD_1^{(2)}$, then 
				\begin{align*}
					(h \circ_{\MD} g) \circ_{\MD} f 
					& = (h \circ_{\MC} g) \circ_{\MC} f \\
					& = h \circ_{\MC} (g \circ_{\MC} f) \\
					& = h \circ_{\MD} (g \circ_{\MD} f).
				\end{align*}  
				\item \tbf{(Identity:)} \\
				Let $X \in \MD_0$. Since $\id_X^{\MC} \in \MD_1$,  
				\begin{align*}
					\dom_{\MD}(\id_X^{\MC}) 
					& = \dom_{\MC}(\id_X^{\MC}) \\
					& = X,
				\end{align*}
				\begin{align*}
					\cod_{\MD}(\id_X^{\MC}) 
					& = \cod_{\MC}(\id_X^{\MC}) \\ 
					& = X,
				\end{align*}
				and for each $f, g \in \MD_1$, if $\dom_{\MD}(f) = X$ and $\cod_{\MD}(g) = X$, then

					f \circ_{\MD} \id_X^{\MC}
					& = f \circ_{\MC} \id_X^{\MC} \\
					& = f
				\end{align*}
				and 
				\begin{align*}
					\id_X^{\MC} \circ_{\MD} g 
					& = id_X^{\MC} \circ_{\MC} g \\
					& = g.
				\end{align*} 
			\end{enumerate}
			Hence $\MD$ is a category.
			\item Clear by construction.
		\end{enumerate}
	\end{proof}
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	\subsection{Product Categories}
	
	\begin{defn} \ld{def:basics:contructs:products:0001} \tcr{change reference to def:basics:constructs:products:0001}
		Let $\MC$ and $\MD$ be categories. We define the \textbf{product category of $\MC$ and $\MD$}, denoted $\MC \times \MD$ by 
		\begin{itemize}
			\item $\Obj(\MC \times \MD) \defeq \{(A, B): A \in \Obj(\MC) \text{ and } B \in \Obj(\MD)\}$
			\item for each $(A, A'), (B, B') \in \Obj(\MC \times \MD)$, 
			$$\Hom_{\MC \times \MD}((A, A'), (B, B')) \defeq \{(f,f'): f \in \Hom_{\MC}(A, B) \text{ and }  f' \in \Hom_{\MC}(A', B') \}$$
			\item for each  $(A, A'), (B, B'), (C, C') \in \Obj(\MC \times \MD)$, $(f, f') \in \Hom_{\MC \times \MD}((A, A'), (B, B'))$ and $(g, g') \in \Hom_{\MC \times \MD}((B, B'), (C, C'))$, 
			$$(g,g') \circ_{\MC \times \MD} (f, f') \defeq  (g \circ_{\MC} f, g' \circ_{\MD} f')$$
		\end{itemize}
	\end{defn}
	
	\begin{ex} \lex{ex:basics:contructs:products:0002}
		Let $\MC$ and $\MD$ be categories. Then $\MC \times \MD$ is a category. 
	\end{ex}
	
	\begin{proof}\
		\begin{itemize}
			\item \textbf{well-definedness of composition: } \\ 
			Let $(A, A'), (B, B'), (C, C') \in \Obj(\MC \times \MD)$, $(f, f') \in \Hom_{\MC \times \MD}((A, A'), (B, B'))$ and $(g, g') \in \Hom_{\MC \times \MD}((B, B'), (C, C'))$. Then $f \in \Hom_{\MC}(A, B)$, $g \in \Hom_{\MC}(B, C)$, $f'  \in \Hom_{\MD}(A', B')$, and  $g' \in \Hom_{\MD}(B', C')$. Hence $g \circ_{\MC} f \in \Hom_{\MC}(A, C)$ and $g' \circ_{\MD} f' \in \Hom_{\MD}(A', C')$. Thus 
			\begin{align*}
				(g,g') \circ_{\MC \times \MD} (f, f') 
				&= (g \circ_{\MC} f, g' \circ_{\MD} f') \\
				& \in \Hom_{\MC \times \MD}((A, A'), (C, C'))
			\end{align*}
			Thus, composition is well defined. \\
			\item \textbf{associativity of composition:} \\
			Let $(A, A'), (B, B'), (C, C'), (D, D') \in \Obj(\MC \times \MD)$, $(f, f') \in \Hom_{\MC \times \MD}((A, A'), (B, B'))$, $(g, g') \in \Hom_{\MC \times \MD}((B, B'), (C, C'))$ and $(h, h') \in \Hom_{\MC \times \MD}((C, C'), (D, D'))$. Then 
			\begin{align*}
				[(h, h') \circ_{\MC \times \MD} (g , g') ] \circ_{\MC \times \MD} (f, f')
				& = (h \circ_{\MC} g, h' \circ_{\MD} g') \circ_{\MC \times \MD} (f, f') \\
				& = ((h \circ_{\MC} g) \circ_{\MC} f, (h' \circ_{\MD} g') \circ_{\MD} f') \\
				& = (h \circ_{\MC} ( g \circ_{\MC} f), h' \circ_{\MD} (g' \circ_{\MD} f')) \\
				& = (h, h') \circ_{\MC \times \MD} (g \circ_{\MC} f, g' \circ_{\MD} f') \\
				& = (h, h') \circ_{\MC \times \MD} [(g, g') \circ_{\MC \times \MD} (f, f')]
			\end{align*} 
			Thus composition is associative. \\
			\item \textbf{existence of identities: } \\
			Let $(A, B) \in \Obj(\MC \times \MD)$, $(f, f'), (g,g') \in \Hom_{\MC \times \MD}$. Suppose that $\dom_{\MC \times \MD}(f, f') = (A, B)$ and $\cod_{\MC \times \MD}(g, g') = (A, B)$. Then $\dom_{\MC}(f) = A$, $\dom_{\MD}(f') = B$, $\cod_{\MC}(g) = A$ and  $\cod_{\MD}(g') = B$. Hence
			\begin{align*}
				(f, f') \circ_{\MC \times \MD} (\id_A^{\MC}, \id_{B}^{\MD}) 
				& = (f \circ_{\MC} \id_A^{\MC}, f' \circ_{\MD} \id_{B}^{\MD}) \\
				& = (f, f)
			\end{align*}
			and
			\begin{align*}
				(\id_A^{\MC}, \id_{B}^{\MD})  \circ_{\MC \times \MD} (g, g')
				& = (\id_A^{\MC} \circ_{\MC} g , \id_{B}^{\MD} \circ_{\MD} g') \\
				& = (g, g')
			\end{align*}
			Therefore $\id_{(A, B)}^{\MC \times \MD} = (\id_A^{\MC}, \id_{B}^{\MD})$.
		\end{itemize}
	\end{proof}
	
	\begin{defn} \ld{def:basics:contructs:products:0003}
		Let $\MA, \MB$ be categories, $A_0 \in \Obj(\MA)$ and $B_0 \in \Obj(\MB)$. We define the
		\begin{itemize}
			\item \tbf{embedding functor of $\MA$ into $\MA \times \MB$ at $B_0$}, denoted $\iota_{\MA}^{B_0}: \MA \rightarrow \MA \times \MB$, by 
			\begin{itemize}
				\item $\iota_{\MA}^{B_0}(A) \defeq (A, B_0), \quad A \in \Obj(\MA)$
				\item $\iota_{\MA}^{B_0}(f) \defeq (f, \id_{B_0}^{\MB}), \quad f \in \Hom_{\MA}$
			\end{itemize}
			\item \tbf{embedding functor of $\MB$ into $\MA \times \MB$ at $A_0$}, denoted $\iota_{\MB}^{A_0}: \MB \rightarrow \MA \times \MB$, by 
			\begin{itemize}
				\item $\iota_{\MB}^{A_0}(B) \defeq (A_0, B), \quad B \in \Obj(\MB)$
				\item $\iota_{\MB}^{A_0}(g) \defeq (\id_{A_0}^{\MA}, g), \quad g \in \Hom_{\MB}$
			\end{itemize}
		\end{itemize}
	\end{defn}
	
	\begin{ex} \lex{ex:basics:contructs:products:0004}
		Let $\MA, \MB$ be categories. Then 
		\begin{enumerate}
			\item for each $B_0 \in \Obj(\MA)$, $\iota_{\MA}^{B_0}:\MA \rightarrow \MA \times \MB$ is a functor, 
			\item for each $A_0 \in \Obj(\MB)$, $\iota_{\MB}^{A_0}:\MB \rightarrow \MA \times \MB$ is a functor.
		\end{enumerate}
	\end{ex}
	
	\begin{proof}\
		\begin{enumerate}
			\item 
			\begin{enumerate}
				\item Let $A_1, A_2 \in \Obj(\MA)$ and $f_{12} \in \Hom_{\MA}(A_1, A_2)$. Then $(f_{12}, \id^{\MB}_{B_0}) \in \Hom_{\MA \times \MB}((A_1, B_0), (A_2, B_0))$. Therefore
				\begin{align*}
					\iota_{\MA}^{B_0}(f_{12})
					& = (f_{12}, \id^{\MB}_{B_0}) \\
					& \in \Hom_{\MA \times \MB}((A_1, B_0), (A_2, B_0)) \\
					& = \Hom_{\MA \times \MB}(\iota_{\MA}^{B_0}(A_1), \iota_{\MA}^{B_0}(A_2)).
				\end{align*}
				\item Let $A_1, A_2, A_3 \in \Obj(\MA)$, $f_{12} \in \Hom_{\MA}(A_1, A_2)$ and $f_{23} \in \Hom_{\MA}(A_3, A_3)$. Then 
				\begin{align*}
					\iota_{\MA}^{B_0}(f_{23} \circ_{\MA} f_{12})
					& = (f_{23} \circ_{\MA} f_{12}, \id_{B_0}^{\MB}) \\
					& = (f_{23} \circ_{\MA} f_{12}, \id_{B_0}^{\MB} \circ_{\MB} \id_{B_0}^{\MB}) \\
					& = (f_{23}, \id_{B_0}^{\MB}) \circ_{\MA \times \MB} (f_{12}, \id_{B_0}^{\MB}) \\
					& = \iota_{\MA}^{B_0}(f_{23}) \circ_{\MA \times \MB} \iota_{\MA}^{B_0}(f_{12}).
				\end{align*}
				\item Let $A \in \Obj(\MA)$. Then 
				\begin{align*}
					\iota_{\MA}^{B_0}(\id_A^{\MA})
					& = (\id_A^{\MA}, \id_{B_0}^{\MB}) \\
					& = \id_{(A, B_0)}^{\MA \times \MB} \\
					& = \id_{\iota_{\MA}^{B_0}(A)}^{\MA \times \MB}.
				\end{align*}
			\end{enumerate}
			Thus $\iota_{\MA}^{B_0}: \MA \rightarrow \MA \times \MB$ is a functor.
			\item Similar \tcr{to prev part (ref prev part)}
		\end{enumerate}
	\end{proof}
	
	\begin{defn} \ld{def:basics:contructs:products:0005}
		Let $\MA, \MB, \MD$ be categories, $F: \MA \times \MB \rightarrow \MD$ and $A_0 \in \Obj(\MA)$, $B_0 \in \Obj(\MB)$. We define the 
		\begin{itemize}
			\item \tbf{slice functor of $F$ at $B_0$}, denoted $F(\cdot, B_0): \MA \rightarrow \MD$ by 
			\begin{itemize}
				\item $F(\cdot, B_0)(A) \defeq F(A, B_0), \quad A \in \Obj(\MA)$
				\item $F(\cdot, B_0)(f) \defeq F(f, \id_{B_0}^{\MB}), \quad f \in \Hom_{\MA}$
			\end{itemize} 
			\item \tbf{slice functor of $F$ at $A_0$}, denoted $F(A_0, \cdot): \MB \rightarrow \MD$ 
			\begin{itemize}
				\item $F(A_0, \cdot)(B) \defeq F(A, B), \quad B \in \MB$
				\item $F(A_0, \cdot)(g) \defeq F(\id_{A_0}^{\MA}, g), \quad g \in \Hom_{\MB}$
			\end{itemize} 			
		\end{itemize}
	\end{defn}

		Let $\MA, \MB, \MD$ be categories, $F: \MA \times \MB \rightarrow \MD$. Then 
		\begin{enumerate}

			\item for each $A_0 \in \Obj(\MA)$, $F(A_0, \cdot): \MB \rightarrow \MD$ is a functor.

	\begin{proof}\
		\begin{enumerate}
			\item Let $B_0 \in \Obj(\MB)$.
			\begin{enumerate}
				\item Let $A_1, A_2 \in \Obj(\MA)$ and $f_{12} \in \Hom_{\MA}(A_1, A_2)$. Then $(f_{12}, \id_{B_0}^{\MB}) \in \Hom_{\MA \times \MB}((A_1, B_0), (A_2, B_0))$. Since $F:\MA \times \MB \rightarrow \MD$, we have that

					& = F(f_{12}, \id_{B_0}^{\MB}) \\
					& \in \Hom_{\MD}(F(A_1, B_0), F(A_2, B_0)) \\
					& = \Hom_{\MD}(F(\cdot, B_0)(A_1), F(\cdot, B_0)(A_2)) \\

			\end{enumerate}
			Thus $F(\cdot, B_0):\MA \rightarrow \MD$ is a functor.

		\end{enumerate}
	\end{proof}

	\subsection{Isomorphisms}
	
	\begin{defn} \ld{15000.1}
		Let $\MC$ be a category and $A \in \Obj(\MC)$.
		\begin{itemize}
			\item Let $f \in \Hom_{\MC}(A, A)$. Then $f$ is said to be an \textbf{endomorphism of $A$}.
			\item We define $\Endo_{\MC}(A) \defeq \Endo_{\MC}(A) = \Hom_{\MC}(A, A)$. 
		\end{itemize}
	\end{defn}
	
	\begin{defn} \ld{15002}
		Let $\MC$ be a category and $A,B \in \Obj(\MC)$.  
		\begin{itemize}
			\item Let $f \in \Hom_{\MC}(A, B)$. Then $f$ is said to be an \textbf{isomorphism} if there exists $g \in \Hom_{\MC}(B, A)$ such that $g \circ f = \id_{A}$ and $f \circ g = \id_{B}$.
			\item We define $\Iso_{\MC}(A, B) \defeq \{f \in \Hom_{\MC}(A,B): f \text{ is an isomorphism}\}$.
		\end{itemize}
	\end{defn}
	
	\begin{defn} \ld{15002.1}
		Let $\MC$ be a category and $A \in \Obj(\MC)$. 
		\begin{itemize}
			\item Let $f \in \Endo_{\MC}(A)$. Then $f$ is said to be an \textbf{automorphism} if $f \in \Iso_{\MC}(A, A)$.
			\item We define $\Aut(A) \defeq \{f \in \Endo_{\MC}(A): f \text{ is an automorphism}\}$.
		\end{itemize}
	\end{defn}
	
	\begin{ex} \lex{15003} \textbf{Uniqueness of Inverses:} \\
		Let $\MC$ be a category, $A,B \in \Obj(\MC)$ and $f \in \Iso_{\MC}(A, B)$. Then there exists a unique $g \in \Iso_{\MC}(B, A)$ such that $g \circ f = \id_A$ and $f \circ g = \id_B$.
	\end{ex}
	
	\begin{proof}\
		\begin{itemize}
			\item \tbf{Existence:} \\
			By definition, since $f$ is an isomorphism, there exists $g \in \Hom_{\MC}(B, A)$ such that $g \circ f = \id_{A}$ and $f \circ g = \id_B$. By definition, $g \in \Iso_{\MC}(B, A)$.
			\item \tbf{Uniqueness: } \\
			Let $g' \in \Iso_{\MC}(B, A)$. Suppose that $g' \circ f = \id_A$, $f \circ g' = \id_B$. Then 
			\begin{align*}
				g'
				& = g' \circ \id_{B} \\
				& = g' \circ (f \circ g) \\
				& = (g' \circ f) \circ  g \\
				& = \id_A \circ g \\
				& = g
			\end{align*}
		\end{itemize}
	\end{proof}
	
	\begin{defn} \ld{15003.1}
		Let $\MC$ be a category, $A,B \in \Obj(\MC)$ and $f \in \Hom_{\MC}(A, B)$. Suppose that $f$ is an isomorphism. We define the \textbf{inverse of $f$}, denoted $f^{-1} \in \Iso_{\MC}(B, A)$, to be the unique $g \in \Iso_{\MC}(B, A)$ such that $g \circ f = \id_A$ and $f \circ g = \id_B$.
	\end{defn}
	
	\begin{ex} \lex{15004}
		Let $\MC$ be a category and $A \in \Obj(\MC)$. Then $\id_A \in \Aut_{\MC}(A)$ and $(\id_A)^{-1} = \id_A$. 
	\end{ex}
	
	\begin{proof}
		Since $\id_A \circ \id_A = \id_A$, we have that $\id_A \in \Aut_{\MC}(A)$ and $(\id_A)^{-1} = \id_A$. 
	\end{proof}
	
	\begin{ex} \lex{15005}
		Let $\MC$ be a category and $A, B \in \Obj(\MC)$. Then for each $f \in \Iso_{\MC}(A, B)$, $(f^{-1})^{-1} = f$.
	\end{ex}
	
	\begin{proof}
		Let $f \in \Iso_{\MC}(A, B)$. By definition, $f \circ f^{-1} = \id_{B}$ and $f^{-1} \circ f = \id_A$. Hence $(f^{-1})^{-1} = f$.
	\end{proof}
	
	\begin{ex} \lex{15006}
		Let $\MC$ be a category and $A,B,C \in \Obj(\MC)$. Then for each $f \in \Iso_{\MC}(A, B)$ and $g \in \Iso_{\MC}(B, C)$, $g \circ f \in \Iso_{\MC}(A, C)$ and $(g \circ f)^{-1} = f^{-1} \circ g^{-1}$. 
	\end{ex}
	
	\begin{proof}
		Let $f \in \Iso_{\MC}(A, B)$ and $g \in \Iso_{\MC}(B, C)$. Then 
		\begin{align*}
			(f^{-1} \circ g^{-1}) \circ (g \circ f) 
			& = ((f^{-1} \circ g^{-1}) \circ g) \circ f \\
			& = (f^{-1} \circ (g^{-1} \circ g)) \circ f \\
			& = (f^{-1} \circ \id_B) \circ f \\
			& = f^{-1} \circ f \\
			& = \id_{A}
		\end{align*}
		and 
		\begin{align*}
			(g \circ f) \circ (f^{-1} \circ g^{-1}) 
			& = ((g \circ f) \circ f^{-1}) \circ g^{-1} \\
			& = (g \circ (f \circ f^{-1})) \circ g^{-1} \\
			& = (g \circ \id_{B}) \circ g^{-1} \\
			& = g \circ  g^{-1} \\
			& = \id_{C}.
		\end{align*}
		Hence $g \circ f \in  \Iso_{\MC}(A, C)$ and $(g \circ f)^{-1} = f^{-1} \circ g^{-1}$.
	\end{proof}
	
	\begin{defn} \ld{15007}
		Let $\MC$ be a category. 
		\begin{itemize}
			\item Let $A,B \in \Obj(\MC)$. Then $A$ is said to be \textbf{isomorphic} to $B$ if $\Iso_{\MC}(A, B) \neq \varnothing$.
			\item We define the \tbf{isomorphism class relation on $\Obj(\MC)$}, denoted ${\cong_{\MC}} \in \MR(A, B)$, by $A \cong_{\MC} B$ iff $A$ is isomorphic to $B$. 
		\end{itemize}
	\end{defn}
	
	\begin{note}
		When the context is clear, we may write $\cong$ in place of $\cong_{\MC}$.
	\end{note}
	
	\begin{ex} \lex{15008}
		Let $\MC$ be a category. Then ${\cong} \in \Equi(\Obj(C))$ \tcr{does $\MC$ neet to be small or just locally small?}.
	\end{ex}
	
	\begin{proof} \
		\begin{enumerate}
			\item \textbf{reflexivity: } \\
			Let $A \in \Obj(\MC)$. \rex{15004} implies that $\id_A \in \Aut(A)$. Thus $A \cong A$. Since $A \in \Obj(\MC)$ is arbitrary, we have that for each $A \in \Obj(\MC)$, $A \cong A$ and thus $\cong$ is reflexive. 
			\item \textbf{symmetry: } \\
			Let $A, B \in \Obj(\MC)$. Suppose that $A \cong B$. Then there exists $f \in \Iso_{\MC}(A,B)$. \rex{15005} implies that $f^{-1} \in \Iso_{\MC}(B, A)$. Hence $B \cong A$.  Since $A, B \in \Obj(\MC)$ with $A \cong B$ are arbitrary, we have that for each $A, B \in \Obj(\MC)$, $A \cong B$ implies that $B \cong A$ and thus $\cong$ is reflexive. 
			\item \textbf{transitivity: }
			Let $A, B, C \in \Obj(\MC)$. Suppose that $A \cong B$ and $B \cong C$. Then there exist $f \in \Iso_{\MC}(A,B)$ and $g \in \Iso_{\MC}(B, C)$. \rex{15006} implies that $g \circ f \in \Iso_{\MC}(A, C)$. Thus $A \cong C$. Since $A, B, C \in \Obj(\MC)$ with $A \cong B$ and $B \cong C$ are arbitrary, we have that for each $A, B, C \in \Obj(\MC)$, $A \cong B$ and $B \cong C$ implies that $A \cong C$ and thus $\cong$ is transitive. 
		\end{enumerate}
		Thus ${\cong} \in \Equi(\Obj(C))$.
	\end{proof}
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	\subsection{Natural Isomorphisms}
	
	\begin{defn} \ld{15011.1}
		Let $\MC$ and $\MD$ be categories, $F,G : \MC \rightarrow \MD$ and $\al: F \Rightarrow G$. Then $\al$ is said to be a \textbf{natural isomorphism} if for each $A \in \Obj(\MC)$, $\al_A \in \Iso_{\MD}(F(A), G(A))$. 
	\end{defn}
	
	\begin{defn} \ld{15011.2}
		Let $\MC$ and $\MD$ be categories, $F,G: \MC \rightarrow \MD$ and $\al : F \Rightarrow G$. Suppose that $\al$ is a natural isomorphism. We define $\al^{-1}: G \Rightarrow F$ by $(\al^{-1})_A = \al_A^{-1}$. 
	\end{defn}
	
	\begin{ex} \lex{15011.3}
		Let $\MC$ and $\MD$ be categories, $F,G: \MC \rightarrow \MD$ and $\al : F \Rightarrow G$. Suppose that $\al$ is a natural isomorphism. Then $\al^{-1}: G \Rightarrow F$ is a natural transformation
	\end{ex}
	
	\begin{proof}\
		\begin{enumerate}
			\item Let $A \in \Obj(\MC)$. Since $\al_A \in \Hom_{\MD}(F(A), G(A))$, we have that 
			\begin{align*}
				(\al^{-1})_A
				& = \al_A^{-1} \\
				& \in \Hom_{\MD}(G(A), F(A))
			\end{align*}
			\item Let $A, B \in \Obj(\MC)$ and $f \in \Hom_{\MC}(A,B)$. Since $G(f) \circ \al_A = \al_B \circ F(f)$, i.e. the following diagram commutes:
			\[ 
			\begin{tikzcd}
				F(A)  \arrow[r, "\al_A"]  \arrow[d, "F(f)"']  & G(A)   \arrow[d, "G(f)"]\\
				F(B) \arrow[r, "\al_B"] &  G(B) \\
			\end{tikzcd}
			\]
			we have that 
			\begin{align*}
				F(f) \circ (\al^{-1})_A 
				& = F(f) \circ \al_A^{-1} \\
				& = \id_{F(B)} \circ (F(f) \circ \al_A^{-1}) \\
				& = (\al_B^{-1} \circ \al_B) \circ (F(f) \circ \al_A^{-1}) \\
				& = \al_B^{-1} \circ (\al_B \circ (F(f) \circ \al_A^{-1})) \\
				& = \al_B^{-1} \circ ((\al_B \circ F(f)) \circ \al_A^{-1}) \\
				& = \al_B^{-1} \circ ((G(f) \circ \al_A) \circ \al_A^{-1}) \\
				& = \al_B^{-1} \circ (G(f) \circ (\al_A \circ \al_A^{-1})) \\
				& = \al_B^{-1} \circ (G(f) \circ \id_{G(A)}) \\
				& = \al_B^{-1} \circ G(f) \\
				& = (\al^{-1})_B \circ G(f) 
			\end{align*}
			i.e. the following diagram commutes:
			\[ 
			\begin{tikzcd}
				G(A)  \arrow[r, "(\al^{-1})_A"]  \arrow[d, "G(f)"']  & F(A)   \arrow[d, "F(f)"]\\
				G(B) \arrow[r, "(\al^{-1})_B"] &  F(B) \\
			\end{tikzcd}
			\]
			So $\al^{-1}: G \Rightarrow F$.
		\end{enumerate}
	\end{proof}
	
	\begin{ex} \lex{15011.4}
		Let $\MC$ and $\MD$ be categories, $F,G: \MC \rightarrow \MD$ and $\al : F \Rightarrow G$. Suppose that $\al$ is a natural isomorphism. Then $\al^{-1} \circ \al = \id_{F}$ and $\al \circ \al^{-1} = \id_G$.
	\end{ex}
	
	\begin{proof}
		Let $A \in \Obj(\MC)$. Then 
		\begin{align*}
			(\al^{-1} \circ \al )_A 
			& = (\al^{-1})_A \circ \al_A \\
			& = \al_A^{-1} \circ \al_A \\
			& = \id_{F(A)} \\
			& = (\id_F)_A
		\end{align*}
		and 
		\begin{align*}
			(\al \circ \al^{-1} )_A 
			& = \al_A \circ (\al^{-1})_A\\
			& = \al_A \circ \al_A^{-1}\\
			& = \id_{G(A)} \\
			& = (\id_G)_A
		\end{align*}
		Since $A \in \Obj(\MC)$ is arbitrary, $\al^{-1} \circ \al = \id_F$ and $\al \circ \al^{-1} = \id_G$.
	\end{proof}
	
	\begin{ex} \lex{15011.5}
		Let $\MC$ and $\MD$ be categories. Suppose that $\MC$ is small. Let $F,G \in \Obj(\MD^{\MC})$ and $\al \in \Hom_{\MD^{\MC}}(F, G)$. Then $\al$ is a natural isomorphism iff $\al \in \Iso_{\MD^{\MC}}(F, G)$. 
	\end{ex}
	
	\begin{proof}\
		\begin{itemize}
			\item $(\implies)$: \\
			Suppose that $\al$ is a natural isomorphism. \rex{15011.4} implies that $\al \in \Iso_{\MD^{\MC}}(F, G)$. 
			\item $(\impliedby)$: \\
			Suppose that $\al \in \Iso_{\MD^{\MC}}(F, G)$. Let $A \in \Obj(\MC)$. Then 
			\begin{align*}
				\al_A \circ (\al^{-1})_A
				& = (\al \circ \al^{-1})_A \\
				& = (\id_{G})_A \\
				& = \id_{G(A)} \\
			\end{align*}
			and similarly, $\al_A^{-1} \circ \al_A = \id_{F(A)}$. Thus $\al_A \in \Iso_{\MD}(F(A), G(A))$. Since $A \in \Obj(\MC)$ is arbitrary, we have that for each $A \in \Obj(\MC)$, $\al_A \in \Iso_{\MD}(F(A), G(A))$. By definition, $\al$ is a natural isomorphism.
		\end{itemize}
	\end{proof}
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	\subsection{Monomorphisms and Epimorphisms}
	
	\begin{defn} \ld{15009}
		Let $\MC$ be a category and $A,B \in \Obj(\MC)$ and $f:A \rightarrow B$. Then 
		\begin{itemize}
			\item 
			$f$ is said to be a \textbf{monomorphism} if for each $C \in \Obj(C)$ and $g,h \in \Hom_{\MC}(C, A)$, $f \circ g = f \circ h$ implies that $g = h$, i.e. we have the following implication of commutative diagrams: 
			\[ 
			\begin{tikzcd}[baseline= 7]
				C \arrow[dr, "g"]&                                   & \\
				& A \arrow[r, "f"] & B  \\
				C \arrow[ur, "h"']&                                   & \\
			\end{tikzcd}
			\implies
			\begin{tikzcd}[baseline= 7]
				C \arrow[bend right=60, swap]{r}{h} \arrow[bend right=-60]{r}{g} & A  \\
			\end{tikzcd}
			\]
			\item 
			$f$ is said to be an \textbf{epimorphism} if for each $C \in \Obj(C)$ and $g,h \in \Hom_{\MC}(B, C)$, $g \circ f = h \circ f$ implies that $g = h$, i.e. we have the following implication of commutative diagrams: \\
			\[ 
			\begin{tikzcd}[baseline= 7]
				&                                   & C \\
				A \arrow[r, "f"]  & B \arrow[ur, "g"] \arrow[dr, "h"'] &   \\
				&                                   & C \\
			\end{tikzcd}
			\implies
			\begin{tikzcd}[baseline= 7]
				B \arrow[bend right=60, swap]{r}{h} \arrow[bend right=-60]{r}{g} & C  \\
			\end{tikzcd}
			\]
		\end{itemize}
	\end{defn}
	
	\begin{defn}
		Let $\MC$ be a category.
		\begin{itemize}
			\item Let $A, B \ in \Obj(\MC)$. We define 
			\begin{itemize}
				\item $\Mono_{\MC}(A,B) \defeq \{f \in \Hom_{\MC}(A,B): \text{$f$ is a monomorphism}\}$.
				\item $\Epi_{\MC}(A,B) \defeq \{f \in \Hom_{\MC}(A,B): \text{$f$ is an epimorphism}\}$.
			\end{itemize}
			\item We define 
			\begin{itemize}
				\item $\Mono_{\MC} \defeq \{f \in \Hom_{\MC}:\text{$f$ is a monomorphism}\}$
				\item $\Epi_{\MC} \defeq \{f \in \Hom_{\MC}:\text{$f$ is a epimorphism}\}$
			\end{itemize}
		\end{itemize}
	\end{defn}
	
	\begin{ex}
		Let $\MC$ be a category and $A,B,C \in \Obj(\MC)$. Then 
		\begin{enumerate}
			\item for each $f \in \Mono_{\MC}(A,B)$ and $g \in \Mono_{\MC}(B,C)$, $g \circ f \in \Mono_{\MC}(A,C)$,
			\item for each $f \in \Epi_{\MC}(A,B)$ and $g \in \Epi_{\MC}(B,C)$, $g \circ f \in \Epi_{\MC}(A,C)$. 
		\end{enumerate}
	\end{ex}
	
	\begin{proof}\
		\begin{enumerate}
			\item Let $f \in \Mono_{\MC}(A,B)$, $g \in \Mono_{\MC}(B,C)$, $X \in \Obj(\MC)$ and $\al,\be \in \Hom_{\MC}(X, A)$. We note that since 
			\begin{align*}
				(g \circ f) \circ \al = (g \circ f) \circ \be
				& \iff g \circ (f \circ \al) = g \circ (f \circ \be) \\
				& \implies f \circ \al = f \circ \be \quad \text{($g \in \Mono_{\MC}$)} \\
				& \implies \al = \be \quad \text{($f \in \Mono_{\MC}$)}.
			\end{align*} 
			Since $X \in \Obj(\MC)$ and $\al,\be \in \Hom_{\MC}(X, A)$ with $(g \circ f) \circ \al = (g \circ f) \circ \be$ are arbitrary, we have that for each $X \in \Obj(\MC)$ and $\al,\be \in \Hom_{\MC}(X, A)$, $(g \circ f) \circ \al = (g \circ f) \circ \be$ implies that $\al = \be$. Hence $g \circ f \in \Mono_{\MC}(A, C)$. Since $f \in \Mono_{\MC}(A,B)$ and $g \in \Mono_{\MC}(B,C)$ are arbitrary, we have that for each $f \in \Mono_{\MC}(A,B)$, $g \in \Mono_{\MC}(B,C)$, $g \circ f \in \Mono_{\MC}(A,C)$.
			\item Let $f \in \Epi_{\MC}(A,B)$, $g \in \Epi_{\MC}(B,C)$, $X \in \Obj(\MC)$ and $\al,\be \in \Hom_{\MC}(C, X)$. We note that since 
			\begin{align*}
				\al \circ (g \circ f) = \be \circ (g \circ f) 
				& \iff (\al \circ g) \circ f = (\be \circ g) \circ f \\
				& \implies \al \circ g = \be \circ g \quad \text{($f \in \Epi_{\MC}$)} \\
				& \implies \al = \be \quad \text{($g \in \Epi_{\MC}$)}.
			\end{align*} 
			Since $X \in \Obj(\MC)$ and $\al,\be \in \Hom_{\MC}(C, X)$ with $\al \circ (g \circ f) = \be \circ (g \circ f)$ are arbitrary, we have that for each $X \in \Obj(\MC)$ and $\al,\be \in \Hom_{\MC}(C, X)$, $\al \circ (g \circ f) = \be \circ (g \circ f)$ implies that $\al = \be$. Hence $g \circ f \in \Epi_{\MC}(A, C)$. Since $f \in \Epi_{\MC}(A,B)$ and $g \in \Epi_{\MC}(B,C)$ are arbitrary, we have that for each $f \in \Epi_{\MC}(A,B)$, $g \in \Epi_{\MC}(B,C)$, $g \circ f \in \Epi_{\MC}(A,C)$.
		\end{enumerate}
	\end{proof}
	
	\begin{ex} \lex{15011}
		Let $\MC$ be a category and $A,B \in \Obj(\MC)$. Then $\Iso_{\MC}(A, B) \subset \Mono_{\MC}(A, B) \cap \Epi_{\MC}(A, B)$. 
	\end{ex}
	
	\begin{proof}
		Let $f \in \Iso_{\MC}(A, B)$. 
		\begin{itemize}
			\item (monomorphism): \\
			Let $C \in \Obj(\MC)$ and $g, h \in \Hom_{\MC}(C, A)$. Suppose that $f \circ g = f \circ h$. Then
			\begin{align*}
				g 
				& = \id_A \circ g \\
				& = (f^{-1} \circ f) \circ g \\
				& = f^{-1} \circ (f \circ g) \\
				& = f^{-1} \circ (f \circ h) \\
				& = (f^{-1} \circ f) \circ h \\
				& = \id_A \circ h \\
				& = h
			\end{align*}
			Since $g, h \in \Hom_{\MC}(C, A)$ with $f \circ g = f \circ h$ are arbitrary, we have that for each $g, h \in \Hom_{\MC}(C, A)$, $f \circ g = f \circ h$ implies that $g = h$. Therefore $f \in \Mono_{\MC}(A, B)$.
			\item  (epimorphism): \\
			Let $C \in \Obj(\MC)$ and $g, h \in \Hom_{\MC}(B, C)$. Suppose that $ g \circ f = h \circ f$. Then
			\begin{align*}
				g 
				& = g \circ \id_B \\
				& = g \circ (f \circ f^{-1}) \\
				& = (g \circ f) \circ f^{-1} \\
				& = (h \circ f) \circ f^{-1} \\
				& = h \circ (f \circ f^{-1}) \\
				& = h \circ \id_B \\
				& = h
			\end{align*}
			Since $g, h \in \Hom_{\MC}(B, C)$ with $g \circ f = h \circ f$ are arbitrary, we have that for each $g, h \in \Hom_{\MC}(B, C)$, $g \circ f = h \circ f$ implies that $g = h$. Therefore $f \in \Epi_{\MC}(A, B)$. 
		\end{itemize}
		Thus $f \in \Mono_{\MC}(A, B) \cap \Epi_{\MC}(A, B)$. Since $f \in \Iso_{\MC}(A, B)$ is arbitrary, we have that for each $f \in \Iso_{\MC}(A, B)$, $f \in \Mono_{\MC}(A, B) \cap \Epi_{\MC}(A, B)$. Hence $\Iso_{\MC}(A, B) \subset \Mono_{\MC}(A, B) \cap \Epi_{\MC}(A, B)$.
	\end{proof}

Hence $f \in \Epi_{\op{\MC}}(B, A)$.

	\subsection{Fundamental Examples}
	
	\begin{ex} \lex{15010}
		\tcr{mono/epi in $\Set$}
		Let $A, B \in \Obj(\Set)$ and $f \in \Hom_{\Set}(A,B)$. Then 
		\begin{enumerate}
			\item $f$ is a monomorphism iff $f$ is injective
			\item $f$ is an epimorphism iff $f$ is surjective 
		\end{enumerate} 
		\tbf{Hint: } consider $C = \{0\}$ and $C = \{0,1\}$.
	\end{ex}
	
	\begin{proof}\
		\begin{enumerate}
			\item Suppose that $f$ is injective. Let $C \in \Obj(\Set)$ and $g,h \in \Hom_{\Set}(C, A)$. Suppose that $f \circ g = f \circ h$. Let $x \in C$. Then $f(g(x)) = f(h(x))$. Injectivity of $f$ implies that $g(x) = h(x)$. Since $x \in C$ is arbitrary, $g = h$. Hence $f$ is a monomorphism. \\
			Conversely, suppose that $f$ is a monomorphism. Let $a, b \in A$. Suppose that $f(a) = f(b)$. Set $C = \{0\}$ and define $g,h: C \rightarrow A$ by $g(0) = a$ and $h(0) = b$. Then  
			\begin{align*}
				f \circ g(0) 
				& = f(g(0)) \\
				& = f(a) \\
				& = f(b) \\
				& = f(h(0)) \\
				& = f \circ h(0) 
			\end{align*}
			Therefore $f \circ g = f \circ h$. Since $f$ is a monomorphism, we have that $g = h$. Hence 
			\begin{align*}
				a 
				& = g(0) \\
				& = h(0) \\
				& = b
			\end{align*}
			\item Suppose that $f$ is surjective. Let $C \in \Obj(\MC)$ and $g,h \in \Hom_{\Set}(B, C)$. Suppose that $g \circ f = h \circ f$. Let $y \in B$. Surjective of $f$ implies that there exists $x \in A$ such that $y = f(x)$. 
			Then 
			\begin{align*}
				g(y)
				& = g(f(x)) \\
				& = g \circ f (x) \\
				& = h \circ f (x) \\
				& = h(f(x)) \\
				& = h(y) 
			\end{align*}
			Since $y \in B$ is arbitrary, $g = h$. Hence $f$ is an epimorphism. \\
			Conversely, suppose that $f$ is an epimorphism. Set $C = \{0,1\}$ and define $g, h: B \rightarrow C$ by $g = \chi_{f(A)}$ and $h = \chi_{B}$. Then $g \circ f = h \circ f$. Since $f$ is an epimorphism, $g = h$ and $f(A) = B$. Hence $f$ is surjective.  
		\end{enumerate}
	\end{proof}

	\section{Initial and Final Objects}
	
	\begin{defn} \ld{15012}
		Let $\MC$ be a category and $0 \in \Obj(\MC)$ and $1 \in \Obj(\MC)$. Then 
		\begin{itemize}
			\item $0$ is said to be \textbf{$\MC$-initial} if for each $A \in \Obj(\MC)$, there exists $f \in \Hom_{\MC}(0, A)$ such that $\Hom_{\MC}(0, A) = \{f\}$. 
			\item $1$ is said to be \textbf{$\MC$-terminal} if for each $A \in \Obj(\MC)$, there exists $f \in \Hom_{\MC}(A, 1)$ such that $\Hom_{\MC}(A, 1) = \{f\}$. \tcr{redo, use duality}
		\end{itemize}
	\end{defn}
	
	\begin{note}
		When the context is clear, we say ``initial" and ``terminal" in place of ``$\MC$-initial" and ``$\MC$-terminal". 
	\end{note}
	
	\begin{ex} \lex{15012.1}
		Let $\MC$ be a category and $0 \in \Obj(\MC)$ and $1 \in \Obj(\MC)$. Then 
		\begin{enumerate}
			\item $0$ is $\MC$-initial iff $0$ is $\op{\MC}$-terminal
			\item $1$ is $\MC$-terminal iff $1$ is $\op{\MC}$-initial
		\end{enumerate}
	\end{ex}
	
	\begin{proof}\
		\begin{enumerate}
			\item 
			\begin{itemize}
				\item $(\implies)$: \\
				Suppose that $0$ is $\MC$-initial. Let $A \in \Obj(\op{\MC})$. Since $\Obj(\op{\MC}) = \Obj(\MC)$, we have that $A \in \Obj(\MC)$. Since $0$ is $\MC$-initial, there exists $f \in \Hom_{\MC}(0,A)$ such that $\Hom_{\MC}(0,A) = \{f\}$. Then 
				\begin{align*}
					\Hom_{\op{\MC}}(A, 0) 
					& = \Hom_{\MC}(0, A) \\
					& = \{f\}.
				\end{align*}
				Since $A$ is arbitrary, we have that for each $A \in \Obj(\op{\MC})$, there exists $f \in \Hom_{\op{\MC}}(A, 0)$ such that $\Hom_{\op{\MC}}(A, 0) = \{f\}$. Thus $0$ is $\op{\MC}$-terminal.
				\item $(\impliedby)$: \\
				Suppose that $0$ is $\op{\MC}$-terminal. Let $A \in \Obj(\MC)$. Since $\Obj(\MC) = \Obj(\op{\MC})$, we have that $A \in \Obj(\op{\MC})$. Since $0$ is $\op{\MC}$-terminal, there exists $f \in \Hom_{\op{\MC}}(A,0)$ such that $\Hom_{\op{\MC}}(0,A) = \{f\}$. Then 
				\begin{align*}
					\Hom_{\MC}(0, A) 
					& = \Hom_{\op{\MC}}(A, 0) 
					& = \{f\}.
				\end{align*}

	\section{Equalizers and Coequalizers}
	
	\begin{defn}
		Let $\MC$ be a category, $A,B, E \in \Obj(\MC)$, $f,g \in \Hom_{\MC}(A,B)$. 
		\begin{itemize}
			\item Let $h \in \Hom_{\MC}(E, A)$. Then $(E,h)$ is said to be an \tbf{equalizer} of $f$ and $g$ if 
			\begin{enumerate}
				\item $f \circ h = g \circ h$, i.e. the following diagram commutes:
				\[ 
				\begin{tikzcd}[baseline= 7]
					&                                   & B \\
					E \arrow[r, "h"]  & A \arrow[ur, "f"] \arrow[dr, "g"'] &   \\
					&                                   & B \\
				\end{tikzcd}
				\]
				\item for each $E' \in \Obj(\MC)$ and $h' \in \Hom_{\MC}(E', A)$, if $f \circ h' = g \circ h'$, then there exists a unique $\phi \in \Hom_{\MC}(E', E)$ such that $h \circ \phi = h'$, i.e. the following diagram commutes:
				\[ 
				\begin{tikzcd}[baseline= 7]
					E' \arrow[dr, "h'"] \arrow[dd, "\phi"', dashed]&     & B \\
					& A \arrow[ur, "f"] \arrow[dr, "g"'] &   \\
					E \arrow[ur, "h"'] &                                  & B \\
				\end{tikzcd}
				\]
			\end{enumerate} 
			\item Let $h \in \Hom_{\MC}(B, E)$. Then $(E,h)$ is said to be a \tbf{coequalizer} of $f$ and $g$ if 
			\begin{enumerate}
				\item $h \circ f = h \circ g$, i.e. the following diagram commutes:
				\[ 
				\begin{tikzcd}[baseline= 7]
					A \arrow[dr, "f"]  &                   &  \\
					& B \arrow[r, "h"]  & E  \\
					A \arrow[ur, "g"']  &                   & \\
				\end{tikzcd}
				\]
				\item for each $E' \in \Obj(\MC)$ and $h' \in \Hom_{\MC}(B, E')$, if $f \circ h' = g \circ h'$, then there exists a unique $\phi \in \Hom_{\MC}(E', E)$ such that $\phi \circ h = h'$, i.e. the following diagram commutes:
				\[ 
				\begin{tikzcd}[baseline= 7]
					A \arrow[dr, "f"]  &                   & E \arrow[dd, "\phi", dashed]\\
					& B \arrow[ur, "h"]  \arrow[dr, "h'"'] &   \\
					A \arrow[ur, "g"']  &                  & E' \\
				\end{tikzcd}
				\]
			\end{enumerate} 
			\tcr{maybe use abrevioation using parallel arrows, try to describe using parallel arrows as a quotient or collaps of a square (for two morphisms) with the same domain and codomain. }
		\end{itemize}
	\end{defn}
	
	
	\begin{ex}
		Let $\MC$ be a category, $A,B,E \in \Obj(\MC)$, $f,g \in \Hom_{\MC}(A,B)$.

			\item Let $h \in \Hom_{\MC}(E, A)$. If $(E, h)$ is an equalizer of $f$ and $g$, then $h \in \Mono_{\MC}(E, A)$. 
			\item Let $h \in \Hom_{\MC}(B, E)$. If $(E, h)$ is a coequalizer of $f$ and $g$, then $h \in \Epi_{\MC}(B, E)$. 
		\end{enumerate}
	\end{ex}
	
	\begin{proof}\
		\begin{enumerate}

				E \arrow[r, "h"]  & A \arrow[ur, "f"] \arrow[dr, "g"'] &   \\

			\begin{align*}
				f \circ h'
				& = f \circ (h \circ \al) \\
				& = (f \circ h) \circ \al \\
				& = (g \circ h) \circ \al \\
				& = g \circ (h \circ \al) \\
				& = g \circ h'.
			\end{align*}
			Since $(E,h)$ is an equalizer of $f$ and $g$, there exists a unique $\phi \in \Hom_{\MC}(E', E)$ such that $h \circ \phi = h'$, i.e. the following diagram commutes:
			\[

				& A \arrow[ur, "f"] \arrow[dr, "g"'] &   \\
				E \arrow[ur, "h"'] &                                  & B \\
			\end{tikzcd}
			\]

			\]

			\item \tcr{FINISH!!!}
		\end{enumerate}
	\end{proof}

	\begin{defn} \ld{def:limits:limits_colimits:0001}
		Let $\MC, \MD$ be a categories, $X \in \Obj(\MD)$, $F: \MC \rightarrow \MD$ and $A \in \Obj(\MC)$.
		\begin{itemize}
			\item Let $f \in \Hom_{\MD}(X, F(A))$. 			\[
			\begin{tikzcd}
				X \arrow[r, "f"]  \arrow[dr, "f'"'] & F(A) \arrow[d, "F(\al)", dashed] \\
				& F(A')  
			\end{tikzcd}
			\hspace{.3cm}
			\begin{tikzcd}
				A  \arrow[d, "\al", dashed] \\
				A'  
			\end{tikzcd}
			\]
			\item Let $f \in \Hom_{\MD}(F(A), X)$. Then $(A, f)$ is said to be a \textbf{universal morphism from $F$ to $X$} if for each $A' \in \Obj(\MC)$ and $f' \in \Hom_{\MD}(F(A'), X)$, there exists a unique $\al \in \Hom_{\MC}(A', A)$ such that $f \circ F(\al) = f'$, i.e. the following diagram commutes:
			\[
			\begin{tikzcd}
				X   & F(A) \arrow[l, "f"'] \\

			\end{tikzcd}

\begin{defn}

		\item $\Hom_{\MC}(\cdot, \cdot)(f_1,f_2)(h) \defeq f_2 \circ_{\MC} h \circ_{\MC} f_1, \quad (f_1, f_2) \in \Hom_{\op{\MC} \times \MC}((A_1, A_2), (B_1, B_2)), h \in \Hom_{\MC}(A_1, A_2)$
	\end{itemize}
\end{defn}

		& \in \Hom_{\Set}(\Hom_{\MC}(A_1, A_2), \Hom_{\MC}(B_1, B_2)) \\
		& = \Hom_{\Set}(H(A_1, A_2), H(B_1, B_2))
	\end{align*}
	\item Let $(A_1, A_2), (B_1, B_2), (C_1, C_2) \in \Obj(\op{\MC} \times \MC)$, $(f_1, f_2) \in \Hom_{\op{\MC} \times \MC}((A_1, A_2), (B_1, B_2))$ and $(g_1, g_2) \in \Hom_{\op{\MC} \times \MC}((B_1, B_2), (C_1, C_2))$. Then for each $h \in \Hom_{\MC}(A_1, A_2)$, we have that 
	\begin{align*}
		H((g_1, g_2) \circ_{\op{\MC} \times \MC}(f_1, f_2))(h)
		& = H(g_1 \circ_{\op{\MC}} f_1, g_2 \circ_{\MC} f_2)(h) \\
		& = H(f_1 \circ_{\MC} g_1, g_2 \circ_{\MC} f_2)(h) \\
		& = (g_2 \circ_{\MC} f_2) \circ_{\MC} h \circ_{\MC} (f_1 \circ_{\MC} g_1) \\
		& = g_2 \circ_{\MC} (f_2 \circ_{\MC} h \circ_{\MC} f_1) \circ_{\MC} g_1 \\
		& = g_2 \circ_{\MC} [H(f_1, f_2)(h)] \circ_{\MC} g_1 \\
		& = H(g_1, g_2)(H(f_1, f_2)(h)) \\
		& = H(g_1, g_2) \circ_{\Set} H(f_1, f_2)(h).
	\end{align*}
	Thus $H((g_1, g_2) \circ_{\op{\MC} \times \MC}(f_1, f_2)) = H(g_1, g_2) \circ H(f_1, f_2)$. Since $(A_1, A_2), (B_1, B_2), (C_1, C_2) \in \Obj(\op{\MC} \times \MC)$, $(f_1, f_2) \in \Hom_{\op{\MC} \times \MC}((A_1, A_2), (B_1, B_2))$ and $(g_1, g_2) \in \Hom_{\op{\MC} \times \MC}((B_1, B_2), (C_1, C_2))$ are arbitrary, we have that for each $(A_1, A_2), (B_1, B_2), (C_1, C_2) \in \Obj(\op{\MC} \times \MC)$, $(f_1, f_2) \in \Hom_{\op{\MC} \times \MC}((A_1, A_2), (B_1, B_2))$ and $(g_1, g_2) \in \Hom_{\op{\MC} \times \MC}((B_1, B_2), (C_1, C_2))$, $H((g_1, g_2) \circ_{\op{\MC} \times \MC}(f_1, f_2)) = H(g_1, g_2) \circ_{\Set} H(f_1, f_2)$. 
	\item Let $(A_1, A_2) \in \Obj(\op{\MC} \times \MC)$. Then for each $h \in \Hom_{\MC}(A_1, A_2)$,
	\begin{align*}
		H(\id_{(A_1, A_2)}^{\op{\MC} \times \MC})(h)
		& = H(\id_{A_1}^{\op{\MC}}, \id_{A_2}^{\MC})(h) \\
		& = H(\id_{A_1}^{\MC}, \id_{A_2}^{\MC})(h) \\
		& = \id_{A_2}^{\MC} \circ_{\MC} h \circ_{\MC} \id_{A_1}^{\MC} \\
		& = h \\
		& = \id_{\Hom_{\MC}(A_1, A_2)}^{\Set}(h) \\
		& = \id_{H(A_1, A_2)}^{\Set}(h).
	\end{align*}
	Thus $H(\id_{(A_1, A_2)}^{\op{\MC} \times \MC}) = \id_{H(A_1, A_2)}^{\Set}$. Since $(A_1, A_2) \in \Obj(\op{\MC} \times \MC)$ is arbitrary, we have that for each $(A_1, A_2) \in \Obj(\op{\MC} \times \MC)$, $\Hom_{\MC}(\id_{(A_1, A_2)}^{\op{\MC} \times \MC}) = \id_{H(A_1, A_2)}^{\Set}$.
	\end{enumerate}
	Hence $\Hom_{\MC}: \op{\MC} \times \MC \rightarrow \Set$ is a functor. 
\end{proof}

\begin{note}
	If $\MC$ is locally small, \rex{ex:basics:contructs:products:0006} implies that for each $X \in \Obj(\MC)$, $\Hom_{\MC}(X, \cdot): \MC \rightarrow \Set$ and $\Hom_{\MC}(\cdot, X): \op{\MC} \rightarrow \Set$ are functors.
\end{note}

\begin{ex}
	Let $\MC$ be a category. Suppose that $\MC$ is locally small. Then 
	\begin{enumerate}
		\item for each $X \in \Obj(\op{\MC})$, $\Hom_{\op{\MC}}(X, \cdot) = \Hom_{\MC}(\cdot, X)$
		\item for each $X \in \Obj(\MC)$, $\Hom_{\MC}(X, \cdot) = \Hom_{\op{\MC}}(\cdot, X)$
	\end{enumerate}
\end{ex}

\begin{proof}
	We note that \rex{} \tcr{(ref ex)} implies that $\op{(\op{\MC})} = \MC$. So $\op{(\op{\MC})} \times \op{\MC} = \MC\times \op{\MC}$.
	\begin{enumerate}
		\item Let $X \in \Obj(\op{\MC})$. Then $X \in \Obj(\MC)$.
		\begin{itemize}
			\item Let $A \in \Obj(\op{\MC})$. Then $A \in \Obj(\MC)$ and
			\begin{align*}
				\Hom_{\op{\MC}}(X, \cdot)(A)
				& = \Hom_{\op{\MC}}(\cdot, \cdot)(X, A) \\
				& = \Hom_{\op{\MC}}(X, A) \\
				& = \Hom_{\MC}(A, X) \\
				& = \Hom_{\MC}(\cdot, \cdot)(A, X) \\
				& = \Hom_{\MC}(\cdot, X)(A). 
			\end{align*}
			\item Let $A,B \in \Obj(\op{\MC})$ and $f \in \Hom_{\op{\MC}}(A, B)$ and $h \in \Hom_{\op{\MC}}(X, A)$. Then $A,B \in \Obj(\MC)$, $f \in \Hom_{\MC}(B,A)$ and $h \in \Hom_{\MC}(A, X)$. Thus $(\id_X^{\MC}, f) \in \Hom_{\MC \times \op{\MC}}((X, A), (X, B))$ and
			\begin{align*}
				\Hom_{\op{\MC}}(X, \cdot)(f)(h)
				& = \Hom_{\op{\MC}}(\cdot, \cdot)(\id_X^{\MC}, f)(h) \\
				& = f \circ_{\op{\MC}} h \circ_{\op{\MC}} \id_X^{\MC} \\
				& = \id_X^{\MC} \circ_{\MC} h \circ_{\MC} f \\
				& = \Hom_{\MC}(\cdot, \cdot)(f, \id_X^{\MC})(h) \\
				& = \Hom_{\MC}(\cdot, X)(f)(h)
			\end{align*}
			Since $h \in \Hom_{\op{\MC}}(X, A)$ is arbitrary, we have that for each $h \in \Hom_{\op{\MC}}(X, A)$, $\Hom_{\op{\MC}}(X, \cdot)(f)(h) = \Hom_{\MC}(\cdot, X)(f)(h)$. Thus $\Hom_{\op{\MC}}(X, \cdot)(f) = \Hom_{\MC}(\cdot, X)(f)$.
		\end{itemize}
		Hence $\Hom_{\op{\MC}}(X, \cdot) = \Hom_{\MC}(\cdot, X)$.
		\item \tcr{FINISH!!!} (similar)
	\end{enumerate}
\end{proof}

\begin{defn}
	Let $\MC$ be a category. Suppose that $\MC$ is locally small. Let $F: \MC \rightarrow \Set$. 
	\begin{itemize}
		\item Let $X \in \Obj(\MC)$ and $\phi : F \Rightarrow \Hom_{\MC}(X, \cdot)$. Then $(X, \phi)$ is said to be a \tbf{representation of $F$} if $\phi$ is a natural isomorphism.
		\item Then $F$ is said to be \tbf{representable} if there exists $X \in \Obj(\MC)$ and $\phi:F \Rightarrow \Hom_{\MC}(X, \cdot)$ such that $(X, \phi)$ is a representation of $F$. 
	\end{itemize}
\end{defn}

\begin{ex}
	Let $\MC$ be a category. Suppose that $\MC$ is small. Let $X \in \Obj(\MC)$. Let $1 \in \Obj(\Set)$. Suppose that $1$ is $\Set$-terminal. Then
	\begin{enumerate}
		\item $X$ is $\MC$-initial iff $\Hom_{\MC}(X, \cdot) \cong_{\Set^{\MC}} \Del^{\MC}_1$.
		\item $X$ is $\MC$-terminal iff $\Hom_{\op{\MC}}(\cdot, X) \cong_{\Set^{\op{\MC}}} \Del^{\op{\MC}}_1$.
	\end{enumerate}
\end{ex}

				& \in \Hom_{\Set}(F(A), 1) \\
				& = \Iso_{\Set}(F(A), 1) \\
				& = \Iso_{\Set}(F(A), \Del^{\MC}_1(A)).
			\end{align*}
			Since $A \in \Obj(\MC)$ is arbitrary, we have that for each $A \in \Obj(\MC)$, $\phi^{(F,1)}_A \in \Iso_{\Set}(F(A), \Del^{\MC}_1(A))$. Thus $\phi^{(F,1)}:F \Rightarrow \Del^{\MC}_1$ is a natural isomorphism. \rex{15011.5} then implies that $\phi^{(F,1)} \in \Iso_{\Set^{\MC}}(F, \Del^{\MC}_1)$. Since $\Iso_{\Set^{\MC}}(F, \Del^{\MC}_1) \neq \varnothing$, we have that $F \cong_{\Set^{\MC}} \Del^{\MC}_1$.
			\item $(\impliedby)$: \\
			Suppose that $F \cong_{\Set^{\MC}} \Del^{\MC}_1$. Then there exists $\phi \in \Iso_{\Set^{\MC}}(F, \Del^{\MC}_1)$. \rex{15011.5} then implies that $\phi: F \Rightarrow \Del^{\MC}_1$ is a natural isomorphism. Let $A \in \Obj(\MC)$. Then 
			\begin{align*}
				\phi_A 
				& \in \Iso_{\Set}(F(A), \Del^{\MC}_1(A)) \\
				& = \Iso_{\Set}(F(A), 1).
			\end{align*}
			Since $\Iso_{\Set}(F(A), 1) \neq \varnothing$ and $1$ is $\Set$-terminal, we have that  
			\begin{align*}
				\Hom_{\MC}(X, A) 
				& = F(A) \\
				& \cong_{\Set} 1 \\
				& \cong_{\Set} \{\varnothing\}, \quad \text{(by \rex{15015})}.
			\end{align*}
			Thus there exists $f \in \Hom_{\MC}(X, A)$ such that $\Hom_{\MC}(X, A) = \{f\}$. Since $A \in \Obj(\MX)$ is arbitrary, we have that for each $A \in \Obj(\MC)$, there exists $f \in \Hom_{\MC}(X, A)$ such that $\Hom_{\MC}(X, A) = \{f\}$. Hence $X$ is $\MC$-initial.
		\end{itemize}

			& \iff \Hom_{\MC}(\cdot, X) \cong_{\Set^{\op{\MC}}} \Del^{\op{\MC}}_1, \quad \text{(by \tcr{(ref) prev ex about $\Hom_{\MC}(\cdot, X) = \Hom_{\op{\MC}}(X, \cdot)$})}
		\end{align*}
	\end{enumerate}
\end{proof}

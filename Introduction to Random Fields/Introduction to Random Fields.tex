	\begin{defn}
	Let $(\Om, \MF, P)$ be a complete probability space, $(M, \rho)$ a separable, totally bounded metric space, $f: \Om \times M \rightarrow \R$. Then $f$ is said to be a \textbf{random field} on $M$.  
	\end{defn}

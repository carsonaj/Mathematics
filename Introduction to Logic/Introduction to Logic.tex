	\subsection{First-Order Languages}
	
	Let $\ML$ be a language. Then $\Symb(\ML)$ is said to be \tbf{first-order} if 
	\begin{enumerate}
		\item ${=, \top, \perp} \in \Symb(\ML)$
		\item $\Var(\ML)$ is countable
		\item $\Conn(\ML) = \{\vee, \wedge, \neg\}$
		\item $\Quan(\ML) = \{\exists, \forall\}$
		\item $\Aux(\ML) = \{\text{`})\text{'}, \text{`}(\text{'}, \text{`},\text{'} \}$
		\item $\Fun(\ML)$ is countable
		\item $\Rel(\ML)$ is countable
		\item $\Cons(\ML)$ is countable
	\end{enumerate}

	\begin{note}
		Here $=$ is just a symbol and carries no interpretation of equality.

	\tcr{idea}

	\section{The Fourier Transform on $\R^n$}	


	\subsection{Schwartz Space}
	\begin{defn}
	Let $f \in C^{\infty}(\R^n)$ and $\al, \bet \in \N_0^n$. We define $$\|f\|_{\al, \bet} = \sup_{x \in \R^n} |(1+x^{\bet}) \p^{\al}f (x)|$$
	We define Schwartz space, denoted $\MS$, by $$\MS = \{f \in C^{\infty}(\R^n): \text{ for each $\al$, $\bet \in \N_0^n$, } \|f\|_{\al, \bet} < \infty\}$$
	\end{defn}
